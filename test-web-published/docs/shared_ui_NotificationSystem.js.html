/**
 * NotificationSystem - Système de notifications toast
 *
 * @module NotificationSystem
 * @version 1.0.0
 */

class NotificationSystem {
  constructor() {
    // Icônes pour chaque type
    this.icons = {
      success: '✓',
      error: '✕',
      warning: '⚠',
      info: 'ℹ'
    };

    // Titres par défaut
    this.titles = {
      success: 'Succès',
      error: 'Erreur',
      warning: 'Attention',
      info: 'Information'
    };
  }

  /**
   * Affiche une notification
   * @param {string} message - Le message à afficher
   * @param {string} type - Type: 'success', 'error', 'warning', 'info'
   * @param {number} duration - Durée en ms (défaut: 3000)
   * @param {string} title - Titre personnalisé (optionnel)
   */
  show(message, type = 'info', duration = 3000, title = null) {
    // Log en console
    if (type === 'error') {
      console.error(message);
    } else {
      console.log(message);
    }

    // Récupère ou crée le conteneur
    let container = document.getElementById('toast-container');
    if (!container) {
      container = document.createElement('div');
      container.id = 'toast-container';
      document.body.appendChild(container);
    }

    // Crée la notification
    const toast = document.createElement('div');
    toast.className = `toast-notification toast-${type}`;

    // Icône
    const icon = document.createElement('div');
    icon.className = 'toast-icon';
    icon.textContent = this.icons[type] || 'ℹ';

    // Contenu
    const content = document.createElement('div');
    content.className = 'toast-content';

    const titleElement = document.createElement('div');
    titleElement.className = 'toast-title';
    titleElement.textContent = title || this.titles[type];

    const messageElement = document.createElement('div');
    messageElement.className = 'toast-message';
    messageElement.textContent = message;

    content.appendChild(titleElement);
    content.appendChild(messageElement);

    // Bouton fermer
    const closeBtn = document.createElement('button');
    closeBtn.className = 'toast-close';
    closeBtn.innerHTML = '×';
    closeBtn.onclick = () => this.remove(toast);

    // Barre de progression
    const progress = document.createElement('div');
    progress.className = 'toast-progress';
    progress.style.animationDuration = `${duration}ms`;

    // Assemblage
    toast.appendChild(icon);
    toast.appendChild(content);
    toast.appendChild(closeBtn);
    toast.appendChild(progress);

    // Ajout au conteneur
    container.appendChild(toast);

    // Fermeture au clic
    toast.onclick = () => this.remove(toast);

    // Auto-suppression
    setTimeout(() => this.remove(toast), duration);
  }

  /**
   * Supprime une notification avec animation
   */
  remove(toast) {
    if (!toast || !toast.parentElement) return;

    toast.classList.add('hiding');
    setTimeout(() => {
      if (toast.parentElement) {
        toast.remove();
      }
    }, 300);
  }

  /**
   * Affiche une notification de succès
   * @param {string} message - Message
   * @param {number} duration - Durée
   * @param {string} title - Titre personnalisé
   */
  success(message, duration = 3000, title = null) {
    this.show(message, 'success', duration, title);
  }

  /**
   * Affiche une notification d'erreur
   * @param {string} message - Message
   * @param {number} duration - Durée
   * @param {string} title - Titre personnalisé
   */
  error(message, duration = 4000, title = null) {
    this.show(message, 'error', duration, title);
  }

  /**
   * Affiche une notification d'avertissement
   * @param {string} message - Message
   * @param {number} duration - Durée
   * @param {string} title - Titre personnalisé
   */
  warning(message, duration = 3500, title = null) {
    this.show(message, 'warning', duration, title);
  }

  /**
   * Affiche une notification d'information
   * @param {string} message - Message
   * @param {number} duration - Durée
   * @param {string} title - Titre personnalisé
   */
  info(message, duration = 3000, title = null) {
    this.show(message, 'info', duration, title);
  }
}

// Instance singleton
const notificationSystem = new NotificationSystem();

export default notificationSystem;
