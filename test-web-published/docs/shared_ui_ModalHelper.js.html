/**
 * ========================================
 * MODAL HELPER - Utilitaire de gestion des modals Bootstrap
 * ========================================
 *
 * Classe utilitaire pour gérer les modals Bootstrap 5.3.3 de manière centralisée.
 * Réduit la duplication de code à travers tous les modules de l'application.
 *
 * Architecture :
 * - **Méthodes statiques** : Pas besoin d'instanciation, utilisation directe
 * - **Gestion d'erreurs** : Logs clairs si modal introuvable
 * - **API simple** : show(), hide(), getInstance(), toggle()
 * - **Délai de transition** : Support pour éviter conflits Bootstrap
 *
 * ==========================================
 * PROBLÈME RÉSOLU
 * ==========================================
 *
 * **Avant ModalHelper** :
 * - Code dupliqué dans 8+ modules
 * - Gestion d'erreurs incohérente
 * - Difficile à maintenir
 *
 * **Après ModalHelper** :
 * - Code centralisé
 * - Gestion d'erreurs uniforme
 * - ~100 lignes de code économisées
 *
 * ==========================================
 * UTILISATION
 * ==========================================
 *
 * @example
 * // Afficher un modal
 * ModalHelper.show('addMaterialModal');
 *
 * @example
 * // Fermer un modal
 * ModalHelper.hide('confirmDeleteModal');
 *
 * @example
 * // Afficher après fermeture d'un autre (évite conflit Bootstrap)
 * ModalHelper.hide('listModal');
 * ModalHelper.showAfterDelay('confirmModal', 300);
 *
 * @example
 * // Obtenir instance pour manipulation avancée
 * const modal = ModalHelper.getInstance('myModal');
 * if (modal) {
 *   modal.toggle();
 * }
 *
 * @class ModalHelper
 * @version 1.0.0
 * @author AutoCalc OptiDevis
 */

class ModalHelper {
  /**
   * Délai standard pour éviter les conflits de transition Bootstrap (ms).
   * Utilisé entre la fermeture d'un modal et l'ouverture d'un autre.
   *
   * @constant {number}
   * @static
   */
  static TRANSITION_DELAY = 300;

  /**
   * Affiche un modal Bootstrap par son ID.
   *
   * Workflow :
   * 1. Récupère l'élément DOM par ID
   * 2. Si introuvable → log erreur + return null
   * 3. Crée instance Bootstrap.Modal
   * 4. Appelle modal.show()
   * 5. Retourne instance pour manipulation ultérieure
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {bootstrap.Modal|null} Instance du modal ou null si introuvable
   *
   * @example
   * const modal = ModalHelper.show('addMaterialModal');
   * if (modal) {
   *   console.log('Modal affiché avec succès');
   * }
   */
  static show(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      console.error(`[ModalHelper] Modal introuvable: #${elementId}`);
      return null;
    }

    try {
      const modal = new bootstrap.Modal(element);
      modal.show();
      return modal;
    } catch (error) {
      console.error(`[ModalHelper] Erreur lors de l'affichage du modal #${elementId}:`, error);
      return null;
    }
  }

  /**
   * Affiche un modal après un délai (pour éviter conflits Bootstrap).
   * Utile lors de la fermeture d'un modal suivie de l'ouverture d'un autre.
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @param {number} [delay=300] - Délai en millisecondes (défaut: 300ms)
   * @returns {Promise<bootstrap.Modal|null>} Promise résolue avec l'instance du modal
   *
   * @example
   * // Workflow typique : liste → confirmation
   * ModalHelper.hide('listModal');
   * await ModalHelper.showAfterDelay('confirmModal', 300);
   */
  static showAfterDelay(elementId, delay = this.TRANSITION_DELAY) {
    return new Promise((resolve) => {
      setTimeout(() => {
        const modal = this.show(elementId);
        resolve(modal);
      }, delay);
    });
  }

  /**
   * Ferme un modal Bootstrap par son ID.
   *
   * Workflow :
   * 1. Récupère l'élément DOM par ID
   * 2. Si introuvable → log erreur + return
   * 3. Récupère instance Bootstrap existante
   * 4. Si instance existe → appelle hide()
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {void}
   *
   * @example
   * ModalHelper.hide('confirmDeleteModal');
   */
  static hide(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      console.error(`[ModalHelper] Modal introuvable: #${elementId}`);
      return;
    }

    try {
      const modal = bootstrap.Modal.getInstance(element);
      if (modal) {
        modal.hide();
      }
    } catch (error) {
      console.error(`[ModalHelper] Erreur lors de la fermeture du modal #${elementId}:`, error);
    }
  }

  /**
   * Obtient l'instance Bootstrap.Modal existante d'un modal.
   * Retourne null si le modal n'existe pas ou n'a pas encore été initialisé.
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {bootstrap.Modal|null} Instance du modal ou null
   *
   * @example
   * const modal = ModalHelper.getInstance('myModal');
   * if (modal) {
   *   // Modal existe et est initialisé
   *   modal.toggle();
   * } else {
   *   // Modal n'existe pas ou pas encore initialisé
   *   ModalHelper.show('myModal');
   * }
   */
  static getInstance(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      console.error(`[ModalHelper] Modal introuvable: #${elementId}`);
      return null;
    }

    return bootstrap.Modal.getInstance(element);
  }

  /**
   * Obtient ou crée une instance Bootstrap.Modal.
   * Si le modal existe déjà, retourne l'instance existante.
   * Sinon, crée une nouvelle instance.
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {bootstrap.Modal|null} Instance du modal ou null
   *
   * @example
   * const modal = ModalHelper.getOrCreateInstance('myModal');
   * modal.show();
   */
  static getOrCreateInstance(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      console.error(`[ModalHelper] Modal introuvable: #${elementId}`);
      return null;
    }

    try {
      return bootstrap.Modal.getOrCreateInstance(element);
    } catch (error) {
      console.error(`[ModalHelper] Erreur lors de la création de l'instance du modal #${elementId}:`, error);
      return null;
    }
  }

  /**
   * Bascule l'affichage d'un modal (affiche si fermé, ferme si affiché).
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {void}
   *
   * @example
   * ModalHelper.toggle('settingsModal');
   */
  static toggle(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      console.error(`[ModalHelper] Modal introuvable: #${elementId}`);
      return;
    }

    try {
      const modal = bootstrap.Modal.getOrCreateInstance(element);
      modal.toggle();
    } catch (error) {
      console.error(`[ModalHelper] Erreur lors du toggle du modal #${elementId}:`, error);
    }
  }

  /**
   * Vérifie si un modal est actuellement affiché.
   *
   * @static
   * @param {string} elementId - ID de l'élément modal (sans #)
   * @returns {boolean} true si le modal est affiché, false sinon
   *
   * @example
   * if (ModalHelper.isShown('confirmModal')) {
   *   console.log('Le modal de confirmation est affiché');
   * }
   */
  static isShown(elementId) {
    const element = document.getElementById(elementId);

    if (!element) {
      return false;
    }

    return element.classList.contains('show');
  }

  /**
   * Ferme tous les modals actuellement affichés.
   * Utile pour nettoyer l'état de l'interface.
   *
   * @static
   * @returns {void}
   *
   * @example
   * // Fermer tous les modals avant navigation
   * ModalHelper.hideAll();
   * navigateToNewPage();
   */
  static hideAll() {
    const modals = document.querySelectorAll('.modal.show');

    modals.forEach((modalElement) => {
      try {
        const modal = bootstrap.Modal.getInstance(modalElement);
        if (modal) {
          modal.hide();
        }
      } catch (error) {
        console.error(`[ModalHelper] Erreur lors de la fermeture du modal:`, error);
      }
    });
  }
}

// Export pour utilisation dans les modules
export default ModalHelper;
