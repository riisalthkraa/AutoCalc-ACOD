/**
 * Calculations - Fonctions de calcul pour devis
 *
 * @module calculations
 * @version 1.0.0
 */

/**
 * Calcule les totaux d'une liste d'items
 * @param {Array} items - Liste des items du devis
 * @returns {Object} { totalHT, totalTVA, totalTTC }
 */
export function calculateTotals(items) {
  if (!Array.isArray(items) || items.length === 0) {
    return {
      totalHT: 0,
      totalTVA: 0,
      totalTTC: 0
    };
  }

  const totalHT = items.reduce((sum, item) => {
    const priceHT = parseFloat(item.priceHT || item.totalHT || item.totalPriceHT || 0);
    return sum + priceHT;
  }, 0);

  const totalTVA = items.reduce((sum, item) => {
    const tva = parseFloat(item.tva || item.totalTVA || item.totalPriceTVA || 0);
    return sum + tva;
  }, 0);

  const totalTTC = items.reduce((sum, item) => {
    const priceTTC = parseFloat(item.priceTTC || item.totalTTC || item.totalPriceTTC || 0);
    return sum + priceTTC;
  }, 0);

  return {
    totalHT: parseFloat(totalHT.toFixed(2)),
    totalTVA: parseFloat(totalTVA.toFixed(2)),
    totalTTC: parseFloat(totalTTC.toFixed(2))
  };
}

/**
 * Calcule la quantité nécessaire selon le type d'unité
 * @param {Object} material - Matériau
 * @param {number} surface - Surface en m²
 * @param {number} thickness - Épaisseur en m
 * @param {number} length - Longueur en m
 * @param {number} units - Nombre d'unités/heures
 * @param {string} selectedUnit - Type d'unité (/h, unités, m², m³, m)
 * @returns {Object} { quantity: number, calculationText: string }
 */
export function calculateQuantity(material, surface, thickness, length, units, selectedUnit) {
  let quantity = 0;
  let calculationText = '';

  if (selectedUnit === '/h') {
    quantity = Math.ceil(units);
    calculationText = `Prestation: ${units} heures → ${quantity} heures (arrondi à l'unité supérieure)`;
  } else if (selectedUnit === 'unités') {
    quantity = Math.ceil(units);
    calculationText = `Unités: ${units} → ${quantity} unités (arrondi à l'unité supérieure)`;
  } else if (selectedUnit === 'm²') {
    quantity = Math.ceil(surface / (material.coverage || 1));
    calculationText = `Surface: ${surface.toFixed(2)} m² ÷ Couverture: ${(material.coverage || 1).toFixed(2)} m² = ${(surface / (material.coverage || 1)).toFixed(2)} unités → ${quantity} unités (arrondi à l'unité supérieure)`;
  } else if (selectedUnit === 'm³') {
    const volume = surface * thickness;

    // Si densité ET poids unitaire sont définis, calcul intelligent
    if (material.density && material.density > 0 &&
        material.unitWeight && material.unitWeight > 0) {
      const totalWeight = volume * material.density;
      quantity = Math.ceil(totalWeight / material.unitWeight);
      calculationText = `Volume: ${volume.toFixed(2)} m³ × Densité: ${material.density.toFixed(0)} kg/m³ = ${totalWeight.toFixed(0)} kg ÷ ${material.unitWeight.toFixed(0)} kg/unité = ${(totalWeight / material.unitWeight).toFixed(2)} → ${quantity} unités (arrondi)`;
    } else {
      // Calcul classique par coverage
      quantity = Math.ceil(volume / (material.coverage || 1));
      calculationText = `Volume: ${surface.toFixed(2)} m² × ${thickness.toFixed(2)} m ÷ Couverture: ${(material.coverage || 1).toFixed(2)} m³ = ${(volume / (material.coverage || 1)).toFixed(2)} unités → ${quantity} unités (arrondi à l'unité supérieure)`;
    }
  } else if (selectedUnit === 'm') {
    quantity = Math.ceil(length / (material.coverage || 1));
    calculationText = `Longueur: ${length.toFixed(2)} m ÷ Couverture: ${(material.coverage || 1).toFixed(2)} m = ${(length / (material.coverage || 1)).toFixed(2)} unités → ${quantity} unités (arrondi à l'unité supérieure)`;
  }

  // Override si piecesPerM2 est défini pour m²
  if (material.piecesPerM2 && surface > 0 && selectedUnit === 'm²') {
    quantity = Math.ceil(surface * material.piecesPerM2);
    calculationText = `Surface: ${surface.toFixed(2)} m² × ${(material.piecesPerM2).toFixed(2)} pièces/m² = ${(surface * material.piecesPerM2).toFixed(2)} unités → ${quantity} unités (arrondi à l'unité supérieure)`;
  }

  return { quantity, calculationText };
}

/**
 * Calcule le prix total d'un item
 * @param {number} quantity - Quantité
 * @param {number} unitPrice - Prix unitaire HT
 * @param {number} tvaRate - Taux de TVA (ex: 20 pour 20%)
 * @returns {Object} { totalHT, totalTVA, totalTTC }
 */
export function calculateItemTotal(quantity, unitPrice, tvaRate) {
  const totalHT = quantity * unitPrice;
  const totalTVA = totalHT * (tvaRate / 100);
  const totalTTC = totalHT + totalTVA;

  return {
    totalHT: parseFloat(totalHT.toFixed(2)),
    totalTVA: parseFloat(totalTVA.toFixed(2)),
    totalTTC: parseFloat(totalTTC.toFixed(2))
  };
}

/**
 * Applique une remise à un montant
 * @param {number} amount - Montant avant remise
 * @param {number} discountPercent - Pourcentage de remise
 * @returns {Object} { amountAfterDiscount, discountAmount }
 */
export function applyDiscount(amount, discountPercent) {
  const discountAmount = amount * (discountPercent / 100);
  const amountAfterDiscount = amount - discountAmount;

  return {
    amountAfterDiscount: parseFloat(amountAfterDiscount.toFixed(2)),
    discountAmount: parseFloat(discountAmount.toFixed(2))
  };
}

export default {
  calculateTotals,
  calculateQuantity,
  calculateItemTotal,
  applyDiscount
};
