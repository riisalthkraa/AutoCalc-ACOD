/**
 * BulkActionsManager - Gestionnaire d'opérations groupées
 * Fournit des fonctionnalités de sélection multiple et d'actions en masse
 */
export class BulkActionsManager {
  constructor(options = {}) {
    this.moduleName = options.moduleName || 'items';
    this.tableId = options.tableId;
    this.checkboxClass = options.checkboxClass || 'bulk-checkbox';
    this.selectAllId = options.selectAllId;
    this.bulkActionsId = options.bulkActionsId;
    this.undoRedoManager = options.undoRedoManager;
    this.onDeleteCallback = options.onDeleteCallback;
    this.onExportCallback = options.onExportCallback;

    this.selectedItems = new Set();
    this.lastCheckedIndex = null;
  }

  /**
   * Initialise les event listeners pour la sélection groupée
   */
  initialize() {
    const selectAllCheckbox = document.getElementById(this.selectAllId);
    if (selectAllCheckbox) {
      selectAllCheckbox.addEventListener('change', (e) => {
        this.selectAll(e.target.checked);
      });
    }

    // Event delegation pour les checkboxes individuelles
    const table = document.getElementById(this.tableId);
    if (table) {
      table.addEventListener('change', (e) => {
        if (e.target.classList.contains(this.checkboxClass)) {
          this.handleCheckboxChange(e);
        }
      });

      // Support Shift+Click pour sélection en série
      table.addEventListener('click', (e) => {
        if (e.target.classList.contains(this.checkboxClass)) {
          this.handleShiftClick(e);
        }
      });
    }

    this.updateBulkActionsUI();
  }

  /**
   * Sélectionne/désélectionne tous les items
   */
  selectAll(checked) {
    const checkboxes = document.querySelectorAll(`.${this.checkboxClass}`);
    checkboxes.forEach(checkbox => {
      checkbox.checked = checked;
      const itemId = checkbox.dataset.id;
      if (checked) {
        this.selectedItems.add(itemId);
        this.highlightRow(checkbox, true);
      } else {
        this.selectedItems.delete(itemId);
        this.highlightRow(checkbox, false);
      }
    });
    this.updateBulkActionsUI();
  }

  /**
   * Gère le changement d'état d'une checkbox individuelle
   */
  handleCheckboxChange(event) {
    const checkbox = event.target;
    const itemId = checkbox.dataset.id;

    if (checkbox.checked) {
      this.selectedItems.add(itemId);
      this.highlightRow(checkbox, true);
    } else {
      this.selectedItems.delete(itemId);
      this.highlightRow(checkbox, false);
    }

    this.updateSelectAllState();
    this.updateBulkActionsUI();
  }

  /**
   * Support pour Shift+Click (sélection en série)
   */
  handleShiftClick(event) {
    if (!event.shiftKey) {
      this.lastCheckedIndex = this.getCheckboxIndex(event.target);
      return;
    }

    const currentIndex = this.getCheckboxIndex(event.target);
    if (this.lastCheckedIndex === null) {
      this.lastCheckedIndex = currentIndex;
      return;
    }

    const checkboxes = Array.from(document.querySelectorAll(`.${this.checkboxClass}`));
    const start = Math.min(this.lastCheckedIndex, currentIndex);
    const end = Math.max(this.lastCheckedIndex, currentIndex);

    for (let i = start; i <= end; i++) {
      const checkbox = checkboxes[i];
      if (checkbox) {
        checkbox.checked = event.target.checked;
        const itemId = checkbox.dataset.id;
        if (checkbox.checked) {
          this.selectedItems.add(itemId);
          this.highlightRow(checkbox, true);
        } else {
          this.selectedItems.delete(itemId);
          this.highlightRow(checkbox, false);
        }
      }
    }

    this.lastCheckedIndex = currentIndex;
    this.updateSelectAllState();
    this.updateBulkActionsUI();
  }

  /**
   * Obtient l'index d'une checkbox dans la liste
   */
  getCheckboxIndex(checkbox) {
    const checkboxes = Array.from(document.querySelectorAll(`.${this.checkboxClass}`));
    return checkboxes.indexOf(checkbox);
  }

  /**
   * Met en surbrillance la ligne sélectionnée
   */
  highlightRow(checkbox, highlight) {
    const row = checkbox.closest('tr');
    if (row) {
      if (highlight) {
        row.classList.add('table-active', 'bulk-selected');
      } else {
        row.classList.remove('table-active', 'bulk-selected');
      }
    }
  }

  /**
   * Met à jour l'état de la checkbox "Select All"
   */
  updateSelectAllState() {
    const selectAllCheckbox = document.getElementById(this.selectAllId);
    if (!selectAllCheckbox) return;

    const checkboxes = document.querySelectorAll(`.${this.checkboxClass}`);
    const checkedCount = this.selectedItems.size;

    if (checkedCount === 0) {
      selectAllCheckbox.checked = false;
      selectAllCheckbox.indeterminate = false;
    } else if (checkedCount === checkboxes.length) {
      selectAllCheckbox.checked = true;
      selectAllCheckbox.indeterminate = false;
    } else {
      selectAllCheckbox.checked = false;
      selectAllCheckbox.indeterminate = true;
    }
  }

  /**
   * Met à jour l'interface des actions groupées
   */
  updateBulkActionsUI() {
    const bulkActionsContainer = document.getElementById(this.bulkActionsId);
    if (!bulkActionsContainer) return;

    const count = this.selectedItems.size;

    if (count === 0) {
      bulkActionsContainer.innerHTML = '';
      bulkActionsContainer.style.display = 'none';
      return;
    }

    bulkActionsContainer.style.display = 'block';
    bulkActionsContainer.innerHTML = `
      <div class="alert alert-primary mb-2 py-2 px-3 d-flex align-items-center justify-content-between" style="background-color: #0d6efd; color: white; border: none;">
        <div class="d-flex align-items-center">
          <i class="fas fa-check-circle me-2"></i>
          <span style="font-size: 0.95rem;"><strong>${count}</strong>&nbsp;${this.getItemLabel(count)} sélectionné(s)</span>
        </div>
        <div class="btn-group btn-group-sm">
          <button class="btn btn-sm btn-light" onclick="window.${this.moduleName}Module.bulkActions.exportSelected()" title="Exporter la sélection">
            <i class="fas fa-download me-1"></i> Exporter
          </button>
          <button class="btn btn-sm btn-danger" onclick="window.${this.moduleName}Module.bulkActions.deleteSelected()" title="Supprimer la sélection">
            <i class="fas fa-trash me-1"></i> Supprimer
          </button>
          <button class="btn btn-sm btn-secondary" onclick="window.${this.moduleName}Module.bulkActions.clearSelection()" title="Annuler la sélection">
            <i class="fas fa-times me-1"></i> Annuler
          </button>
        </div>
      </div>
    `;
  }

  /**
   * Obtient le label approprié pour le type d'item
   */
  getItemLabel(count) {
    const labels = {
      materials: count > 1 ? 'matériaux' : 'matériau',
      clients: 'client(s)',
      suppliers: 'fournisseur(s)',
      quotes: 'devis',
      invoices: count > 1 ? 'factures' : 'facture'
    };
    return labels[this.moduleName] || 'élément(s)';
  }

  /**
   * Exporte les items sélectionnés
   */
  async exportSelected() {
    if (this.selectedItems.size === 0) return;

    if (this.onExportCallback) {
      await this.onExportCallback(Array.from(this.selectedItems));
    }

    this.showNotification(`${this.selectedItems.size} ${this.getItemLabel(this.selectedItems.size)} exporté(s)`, 'success');
  }

  /**
   * Supprime les items sélectionnés
   */
  async deleteSelected() {
    if (this.selectedItems.size === 0) return;

    const count = this.selectedItems.size;
    const label = this.getItemLabel(count);

    const confirmed = await this.showConfirmDialog(
      'Confirmer la suppression',
      `Êtes-vous sûr de vouloir supprimer ${count} ${label} ?`,
      'danger'
    );

    if (!confirmed) return;

    const itemsToDelete = Array.from(this.selectedItems);

    try {
      if (this.onDeleteCallback) {
        const result = await this.onDeleteCallback(itemsToDelete);

        // Enregistrer dans l'historique Undo/Redo si disponible
        if (this.undoRedoManager && result.deletedItems) {
          this.undoRedoManager.recordAction({
            type: 'bulk_delete',
            moduleName: this.moduleName,
            data: result.deletedItems,
            undo: result.undoCallback,
            redo: result.redoCallback
          });
        }
      }

      this.clearSelection();
      this.showNotification(`${count} ${label} supprimé(s) avec succès`, 'success');
    } catch (error) {
      console.error('Erreur lors de la suppression groupée:', error);
      this.showNotification('Erreur lors de la suppression', 'danger');
    }
  }

  /**
   * Annule la sélection
   */
  clearSelection() {
    this.selectedItems.clear();
    const checkboxes = document.querySelectorAll(`.${this.checkboxClass}`);
    checkboxes.forEach(checkbox => {
      checkbox.checked = false;
      this.highlightRow(checkbox, false);
    });
    this.updateSelectAllState();
    this.updateBulkActionsUI();
  }

  /**
   * Obtient les IDs sélectionnés
   */
  getSelectedIds() {
    return Array.from(this.selectedItems);
  }

  /**
   * Affiche une notification
   */
  showNotification(message, type = 'info') {
    // Utiliser le système de notification existant si disponible
    if (window.notificationManager) {
      window.notificationManager.show(message, type);
    } else {
      console.log(`[${type.toUpperCase()}] ${message}`);
    }
  }

  /**
   * Affiche un dialog de confirmation
   */
  showConfirmDialog(title, message, variant = 'primary') {
    return new Promise((resolve) => {
      const confirmed = confirm(`${title}\n\n${message}`);
      resolve(confirmed);
    });
  }

  /**
   * Nettoie les event listeners
   */
  destroy() {
    this.selectedItems.clear();
    this.lastCheckedIndex = null;
  }
}

export default BulkActionsManager;
