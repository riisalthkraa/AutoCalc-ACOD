/**
 * ========================================
 * LOGGER - Système de logging centralisé
 * ========================================
 *
 * Module de logging centralisé pour AutoCalc OptiDevis.
 * Remplace progressivement les console.log dispersés dans l'application
 * pour un contrôle unifié du niveau de log et de la sortie.
 *
 * Avantages :
 * - Niveaux de log hiérarchiques (debug < info < warn < error)
 * - Activation/désactivation par niveau
 * - Formatage uniforme avec timestamps
 * - Identification du module source
 * - Possibilité d'envoyer vers fichier/serveur (futur)
 * - Meilleure lisibilité en développement
 * - Désactivation facile en production
 *
 * Architecture :
 * - Singleton pattern (instance unique)
 * - Configuration globale ou par module
 * - Fallback sur console natif si nécessaire
 *
 * Niveaux de log (ordre hiérarchique) :
 * 1. **DEBUG** (10) - Informations très détaillées pour debugging
 *    Exemple : "Event listener attached", "State updated"
 *
 * 2. **INFO** (20) - Informations générales sur le flux d'exécution
 *    Exemple : "Module initialized", "Quote saved successfully"
 *
 * 3. **WARN** (30) - Avertissements (problème potentiel, mais pas bloquant)
 *    Exemple : "Cache miss", "Deprecated function called"
 *
 * 4. **ERROR** (40) - Erreurs (problème réel nécessitant attention)
 *    Exemple : "Failed to save quote", "Invalid user input"
 *
 * Configuration :
 * Par défaut, niveau = INFO (affiche info, warn, error mais pas debug)
 * En développement : niveau = DEBUG (affiche tout)
 * En production : niveau = ERROR (affiche seulement les erreurs)
 *
 * Usage :
 * ```javascript
 * // Import
 * import Logger from '../shared/utils/Logger.js';
 *
 * // Dans un module
 * class MyModule {
 *   constructor() {
 *     this.logger = Logger.getLogger('MyModule');
 *   }
 *
 *   init() {
 *     this.logger.debug('Initializing...');  // Détails techniques
 *     this.logger.info('Module initialized'); // Info importante
 *     this.logger.warn('Cache miss');         // Avertissement
 *     this.logger.error('Failed to load');    // Erreur
 *   }
 * }
 * ```
 *
 * Configuration globale :
 * ```javascript
 * // Changer le niveau global
 * Logger.setLevel('DEBUG');  // Affiche tout
 * Logger.setLevel('INFO');   // Affiche info, warn, error
 * Logger.setLevel('ERROR');  // Affiche seulement les erreurs
 *
 * // Désactiver complètement
 * Logger.disable();
 *
 * // Réactiver
 * Logger.enable();
 * ```
 *
 * Configuration par module :
 * ```javascript
 * // Activer debug uniquement pour QuotesModule
 * Logger.setLevelForModule('QuotesModule', 'DEBUG');
 * ```
 *
 * @module Logger
 * @version 1.0.0
 * @author AutoCalc OptiDevis Team
 */

/**
 * Niveaux de log disponibles (hiérarchiques)
 * @readonly
 * @enum {number}
 */
const LogLevel = {
  DEBUG: 10,
  INFO: 20,
  WARN: 30,
  ERROR: 40,
  OFF: 100  // Désactive tous les logs
};

/**
 * Mapping nom → niveau
 * @readonly
 * @type {Object<string, number>}
 */
const LogLevelNames = {
  'DEBUG': LogLevel.DEBUG,
  'INFO': LogLevel.INFO,
  'WARN': LogLevel.WARN,
  'ERROR': LogLevel.ERROR,
  'OFF': LogLevel.OFF
};

/**
 * Classe Logger pour un module spécifique
 * @class
 */
class ModuleLogger {
  /**
   * Crée un logger pour un module
   * @param {string} moduleName - Nom du module (ex: "QuotesModule")
   * @param {Logger} parentLogger - Instance parente du Logger
   */
  constructor(moduleName, parentLogger) {
    this.moduleName = moduleName;
    this.parentLogger = parentLogger;
    this.moduleLevel = null; // null = utilise le niveau global
  }

  /**
   * Définit le niveau de log pour ce module spécifiquement
   * @param {string|number} level - Niveau (ex: 'DEBUG', 'INFO')
   */
  setLevel(level) {
    if (typeof level === 'string') {
      this.moduleLevel = LogLevelNames[level.toUpperCase()] ?? LogLevel.INFO;
    } else {
      this.moduleLevel = level;
    }
  }

  /**
   * Obtient le niveau de log actif pour ce module
   * @returns {number}
   */
  getEffectiveLevel() {
    return this.moduleLevel !== null ? this.moduleLevel : this.parentLogger.globalLevel;
  }

  /**
   * Vérifie si un niveau de log est activé
   * @param {number} level - Niveau à vérifier
   * @returns {boolean}
   */
  isLevelEnabled(level) {
    if (!this.parentLogger.enabled) return false;
    return level >= this.getEffectiveLevel();
  }

  /**
   * Formate un message de log avec timestamp et nom de module
   * @param {string} level - Nom du niveau (ex: "DEBUG")
   * @param {any[]} args - Arguments à logger
   * @returns {any[]} Arguments formatés
   */
  formatMessage(level, args) {
    const timestamp = new Date().toISOString().substr(11, 12); // HH:MM:SS.mmm
    const prefix = `[${timestamp}] [${level}] [${this.moduleName}]`;
    return [prefix, ...args];
  }

  /**
   * Log niveau DEBUG
   * @param {...any} args - Arguments à logger
   */
  debug(...args) {
    if (this.isLevelEnabled(LogLevel.DEBUG)) {
      console.log(...this.formatMessage('DEBUG', args));
    }
  }

  /**
   * Log niveau INFO
   * @param {...any} args - Arguments à logger
   */
  info(...args) {
    if (this.isLevelEnabled(LogLevel.INFO)) {
      console.info(...this.formatMessage('INFO', args));
    }
  }

  /**
   * Log niveau WARN
   * @param {...any} args - Arguments à logger
   */
  warn(...args) {
    if (this.isLevelEnabled(LogLevel.WARN)) {
      console.warn(...this.formatMessage('WARN', args));
    }
  }

  /**
   * Log niveau ERROR
   * @param {...any} args - Arguments à logger
   */
  error(...args) {
    if (this.isLevelEnabled(LogLevel.ERROR)) {
      console.error(...this.formatMessage('ERROR', args));
    }
  }

  /**
   * Alias pour debug (compatibilité avec this.log() de BaseModule)
   * @param {...any} args - Arguments à logger
   */
  log(...args) {
    this.debug(...args);
  }
}

/**
 * Classe Logger principale (Singleton)
 * @class
 */
class Logger {
  constructor() {
    if (Logger.instance) {
      return Logger.instance;
    }

    /**
     * Niveau de log global par défaut
     * @type {number}
     */
    this.globalLevel = LogLevel.INFO;

    /**
     * Activation/désactivation globale
     * @type {boolean}
     */
    this.enabled = true;

    /**
     * Cache des loggers de modules
     * @type {Map<string, ModuleLogger>}
     */
    this.moduleLoggers = new Map();

    Logger.instance = this;
  }

  /**
   * Obtient l'instance singleton du Logger
   * @returns {Logger}
   */
  static getInstance() {
    if (!Logger.instance) {
      Logger.instance = new Logger();
    }
    return Logger.instance;
  }

  /**
   * Crée ou récupère un logger pour un module spécifique
   * @param {string} moduleName - Nom du module
   * @returns {ModuleLogger}
   */
  static getLogger(moduleName) {
    const instance = Logger.getInstance();

    if (!instance.moduleLoggers.has(moduleName)) {
      instance.moduleLoggers.set(moduleName, new ModuleLogger(moduleName, instance));
    }

    return instance.moduleLoggers.get(moduleName);
  }

  /**
   * Définit le niveau de log global
   * @param {string|number} level - Niveau (ex: 'DEBUG', 'INFO', 'WARN', 'ERROR', 'OFF')
   */
  static setLevel(level) {
    const instance = Logger.getInstance();

    if (typeof level === 'string') {
      instance.globalLevel = LogLevelNames[level.toUpperCase()] ?? LogLevel.INFO;
    } else {
      instance.globalLevel = level;
    }
  }

  /**
   * Définit le niveau de log pour un module spécifique
   * @param {string} moduleName - Nom du module
   * @param {string|number} level - Niveau
   */
  static setLevelForModule(moduleName, level) {
    const logger = Logger.getLogger(moduleName);
    logger.setLevel(level);
  }

  /**
   * Désactive tous les logs
   */
  static disable() {
    const instance = Logger.getInstance();
    instance.enabled = false;
  }

  /**
   * Réactive les logs
   */
  static enable() {
    const instance = Logger.getInstance();
    instance.enabled = true;
  }

  /**
   * Obtient le niveau de log global actuel
   * @returns {number}
   */
  static getLevel() {
    return Logger.getInstance().globalLevel;
  }

  /**
   * Vérifie si le logger est activé
   * @returns {boolean}
   */
  static isEnabled() {
    return Logger.getInstance().enabled;
  }

  /**
   * Configure le logger selon l'environnement
   * @param {string} env - Environnement ('development', 'production', 'test')
   */
  static configureForEnvironment(env) {
    switch (env) {
      case 'development':
        Logger.setLevel('DEBUG');
        Logger.enable();
        break;
      case 'production':
        Logger.setLevel('ERROR');
        Logger.enable();
        break;
      case 'test':
        Logger.setLevel('WARN');
        Logger.enable();
        break;
      default:
        Logger.setLevel('INFO');
        Logger.enable();
    }
  }
}

// Export du Logger et des niveaux
export default Logger;
export { LogLevel, LogLevelNames };
