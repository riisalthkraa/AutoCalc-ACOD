/**
 * ========================================
 * NOTIFICATION SYSTEM V2 - Enhanced
 * ========================================
 *
 * Système de notifications toast amélioré avec support pour:
 * - Boutons d'action (Annuler, Réessayer, Voir, etc.)
 * - Stack intelligent (max 3 affichés, compteur si plus)
 * - Persistence configurable (errors stay, success auto-dismiss)
 * - Icônes FontAwesome personnalisables
 * - Positions configurables
 *
 * @module NotificationSystemV2
 * @version 2.0.0
 * @author AutoCalc OptiDevis - Phase 1 UX Improvements
 */

class NotificationSystemV2 {
  constructor() {
    // Icônes pour chaque type
    this.icons = {
      success: 'fa-check-circle',
      error: 'fa-exclamation-circle',
      warning: 'fa-exclamation-triangle',
      info: 'fa-info-circle'
    };

    // Titres par défaut
    this.titles = {
      success: 'Succès',
      error: 'Erreur',
      warning: 'Attention',
      info: 'Information'
    };

    // Stack de toasts actifs
    this.activeToasts = [];
    this.maxVisibleToasts = 3;
  }

  /**
   * Affiche une notification avec options avancées
   * @param {string} message - Le message à afficher
   * @param {Object} options - Options de configuration
   * @param {string} options.type - Type: 'success', 'error', 'warning', 'info'
   * @param {number} options.duration - Durée en ms (0 = persistent)
   * @param {string} options.title - Titre personnalisé
   * @param {string} options.icon - Icône FontAwesome personnalisée
   * @param {Object} options.action - Bouton d'action {label, onClick}
   * @param {string} options.position - Position ('top-right', 'top-left', 'bottom-right', 'bottom-left')
   * @param {boolean} options.dismissible - Peut être fermé manuellement
   */
  show(message, options = {}) {
    const {
      type = 'info',
      duration = type === 'error' ? 5000 : 3000,
      title = null,
      icon = null,
      action = null,
      position = 'top-right',
      dismissible = true
    } = options;

    // Log en console
    if (type === 'error') {
      console.error(message);
    } else {
      console.log(message);
    }

    // Récupère ou crée le conteneur
    let container = document.getElementById(`toast-container-${position}`);
    if (!container) {
      container = document.createElement('div');
      container.id = `toast-container-${position}`;
      container.className = `toast-container toast-container-${position}`;
      document.body.appendChild(container);
    }

    // Crée la notification
    const toast = document.createElement('div');
    toast.className = `toast-notification-v2 toast-${type}`;
    toast.dataset.type = type;

    // Icône
    const iconElement = document.createElement('div');
    iconElement.className = 'toast-icon-v2';
    iconElement.innerHTML = `<i class="fas ${icon || this.icons[type]}"></i>`;

    // Contenu
    const content = document.createElement('div');
    content.className = 'toast-content-v2';

    const titleElement = document.createElement('div');
    titleElement.className = 'toast-title-v2';
    titleElement.textContent = title || this.titles[type];

    const messageElement = document.createElement('div');
    messageElement.className = 'toast-message-v2';
    messageElement.textContent = message;

    content.appendChild(titleElement);
    content.appendChild(messageElement);

    // Bouton d'action si fourni
    if (action && action.label && action.onClick) {
      const actionBtn = document.createElement('button');
      actionBtn.className = 'toast-action-btn';
      actionBtn.textContent = action.label;
      actionBtn.onclick = (e) => {
        e.stopPropagation();
        action.onClick();
        this.remove(toast);
      };
      content.appendChild(actionBtn);
    }

    // Bouton fermer (si dismissible)
    if (dismissible) {
      const closeBtn = document.createElement('button');
      closeBtn.className = 'toast-close-v2';
      closeBtn.innerHTML = '<i class="fas fa-times"></i>';
      closeBtn.onclick = (e) => {
        e.stopPropagation();
        this.remove(toast);
      };
      toast.appendChild(closeBtn);
    }

    // Barre de progression (si duration > 0)
    if (duration > 0) {
      const progress = document.createElement('div');
      progress.className = 'toast-progress-v2';
      progress.style.animationDuration = `${duration}ms`;
      toast.appendChild(progress);
    }

    // Assemblage
    toast.appendChild(iconElement);
    toast.appendChild(content);

    // Ajouter au stack
    this.activeToasts.push(toast);
    this.manageStack(container);

    // Ajout au conteneur
    container.appendChild(toast);

    // Animation d'entrée
    setTimeout(() => {
      toast.classList.add('toast-show');
    }, 10);

    // Auto-suppression (si duration > 0)
    if (duration > 0) {
      setTimeout(() => this.remove(toast), duration);
    }

    return toast;
  }

  /**
   * Gère le stack de toasts (max 3 visibles)
   * @param {HTMLElement} container - Conteneur des toasts
   */
  manageStack(container) {
    // Si plus de max toasts, cache les plus anciens
    if (this.activeToasts.length > this.maxVisibleToasts) {
      const excessCount = this.activeToasts.length - this.maxVisibleToasts;

      // Affiche compteur
      this.showStackCounter(container, excessCount);

      // Cache les toasts en excès
      this.activeToasts.slice(0, excessCount).forEach(toast => {
        if (toast && toast.parentElement) {
          toast.style.display = 'none';
        }
      });
    } else {
      this.hideStackCounter(container);
    }
  }

  /**
   * Affiche un compteur de toasts cachés
   * @param {HTMLElement} container - Conteneur
   * @param {number} count - Nombre de toasts cachés
   */
  showStackCounter(container, count) {
    let counter = container.querySelector('.toast-stack-counter');

    if (!counter) {
      counter = document.createElement('div');
      counter.className = 'toast-stack-counter';
      container.insertBefore(counter, container.firstChild);
    }

    counter.textContent = `+${count} notification${count > 1 ? 's' : ''}`;
    counter.style.display = 'flex';
  }

  /**
   * Cache le compteur de stack
   * @param {HTMLElement} container - Conteneur
   */
  hideStackCounter(container) {
    const counter = container.querySelector('.toast-stack-counter');
    if (counter) {
      counter.style.display = 'none';
    }
  }

  /**
   * Supprime une notification avec animation
   * @param {HTMLElement} toast - Élément toast
   */
  remove(toast) {
    if (!toast || !toast.parentElement) return;

    // Retire du stack
    const index = this.activeToasts.indexOf(toast);
    if (index > -1) {
      this.activeToasts.splice(index, 1);
    }

    // Animation de sortie
    toast.classList.add('toast-hiding');
    toast.classList.remove('toast-show');

    setTimeout(() => {
      if (toast.parentElement) {
        toast.remove();

        // Met à jour le stack
        const container = toast.parentElement || document.querySelector('.toast-container');
        if (container) {
          this.manageStack(container);
        }
      }
    }, 300);
  }

  /**
   * Affiche une notification de succès
   * @param {string} message - Message
   * @param {Object} options - Options (duration, title, action, etc.)
   */
  success(message, options = {}) {
    return this.show(message, { ...options, type: 'success' });
  }

  /**
   * Affiche une notification d'erreur
   * @param {string} message - Message
   * @param {Object} options - Options (duration, title, action, etc.)
   */
  error(message, options = {}) {
    return this.show(message, { ...options, type: 'error' });
  }

  /**
   * Affiche une notification d'avertissement
   * @param {string} message - Message
   * @param {Object} options - Options (duration, title, action, etc.)
   */
  warning(message, options = {}) {
    return this.show(message, { ...options, type: 'warning' });
  }

  /**
   * Affiche une notification d'information
   * @param {string} message - Message
   * @param {Object} options - Options (duration, title, action, etc.)
   */
  info(message, options = {}) {
    return this.show(message, { ...options, type: 'info' });
  }

  /**
   * Ferme toutes les notifications
   */
  dismissAll() {
    this.activeToasts.forEach(toast => this.remove(toast));
  }

  /**
   * Ferme toutes les notifications d'un type
   * @param {string} type - Type de notification
   */
  dismissByType(type) {
    this.activeToasts
      .filter(toast => toast.dataset.type === type)
      .forEach(toast => this.remove(toast));
  }
}

// Instance singleton
const notificationSystemV2 = new NotificationSystemV2();

export default notificationSystemV2;
