/**
 * ========================================
 * LOADING STATE MANAGER
 * ========================================
 *
 * Gestionnaire centralisé des états de chargement pour l'application.
 * Fournit skeleton screens, spinners, et progress bars.
 *
 * Fonctionnalités:
 * - Skeleton screens pour tables, cartes, listes
 * - Spinners sur boutons
 * - Progress bars pour opérations longues
 * - États d'erreur avec retry
 *
 * @class LoadingStateManager
 * @version 1.0.0
 * @author AutoCalc OptiDevis - Phase 1 UX Improvements
 */

class LoadingStateManager {
  /**
   * Affiche un skeleton screen dans un conteneur
   * @param {string} selector - Sélecteur CSS du conteneur
   * @param {string} type - Type de skeleton ('table', 'card', 'list')
   * @param {number} rows - Nombre de lignes pour le skeleton
   */
  static showSkeleton(selector, type = 'table', rows = 5) {
    const container = document.querySelector(selector);
    if (!container) {
      console.warn(`[LoadingStateManager] Conteneur introuvable: ${selector}`);
      return;
    }

    // Sauvegarde le contenu original
    if (!container.dataset.originalContent) {
      container.dataset.originalContent = container.innerHTML;
    }

    const skeletons = {
      table: this.createTableSkeleton(rows),
      card: this.createCardSkeleton(),
      list: this.createListSkeleton(rows)
    };

    container.innerHTML = skeletons[type] || skeletons.table;
    container.classList.add('loading-state');
  }

  /**
   * Cache le skeleton et restaure le contenu
   * @param {string} selector - Sélecteur CSS du conteneur
   */
  static hideSkeleton(selector) {
    const container = document.querySelector(selector);
    if (!container) return;

    container.classList.remove('loading-state');

    // Le contenu sera mis à jour par la fonction appelante
    // Ne pas restaurer l'ancien contenu ici
  }

  /**
   * Crée un skeleton pour tableau
   * @param {number} rows - Nombre de lignes
   * @returns {string} HTML du skeleton
   */
  static createTableSkeleton(rows = 5) {
    return `
      <div class="skeleton-table">
        ${Array(rows).fill(0).map(() => `
          <div class="skeleton-row">
            <div class="skeleton-cell skeleton-pulse" style="width: 5%;"></div>
            <div class="skeleton-cell skeleton-pulse" style="width: 35%;"></div>
            <div class="skeleton-cell skeleton-pulse" style="width: 20%;"></div>
            <div class="skeleton-cell skeleton-pulse" style="width: 15%;"></div>
            <div class="skeleton-cell skeleton-pulse" style="width: 15%;"></div>
            <div class="skeleton-cell skeleton-pulse" style="width: 10%;"></div>
          </div>
        `).join('')}
      </div>
    `;
  }

  /**
   * Crée un skeleton pour cartes
   * @returns {string} HTML du skeleton
   */
  static createCardSkeleton() {
    return `
      <div class="skeleton-card">
        <div class="skeleton-card-header skeleton-pulse"></div>
        <div class="skeleton-card-body">
          <div class="skeleton-text skeleton-pulse" style="width: 80%;"></div>
          <div class="skeleton-text skeleton-pulse" style="width: 60%;"></div>
          <div class="skeleton-text skeleton-pulse" style="width: 90%;"></div>
        </div>
      </div>
    `;
  }

  /**
   * Crée un skeleton pour listes
   * @param {number} items - Nombre d'items
   * @returns {string} HTML du skeleton
   */
  static createListSkeleton(items = 5) {
    return `
      <div class="skeleton-list">
        ${Array(items).fill(0).map(() => `
          <div class="skeleton-list-item">
            <div class="skeleton-avatar skeleton-pulse"></div>
            <div class="skeleton-list-content">
              <div class="skeleton-text skeleton-pulse" style="width: 70%;"></div>
              <div class="skeleton-text skeleton-pulse" style="width: 50%;"></div>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }

  /**
   * Affiche un spinner sur un bouton
   * @param {HTMLElement} buttonElement - Élément bouton
   * @param {boolean} loading - État de chargement
   * @param {string} loadingText - Texte pendant le chargement
   */
  static buttonLoading(buttonElement, loading = true, loadingText = 'Chargement...') {
    if (!buttonElement) return;

    if (loading) {
      buttonElement.disabled = true;
      buttonElement.dataset.originalContent = buttonElement.innerHTML;
      buttonElement.innerHTML = `<i class="fas fa-spinner fa-spin"></i> ${loadingText}`;
      buttonElement.classList.add('btn-loading');
    } else {
      buttonElement.disabled = false;
      buttonElement.innerHTML = buttonElement.dataset.originalContent || buttonElement.innerHTML;
      buttonElement.classList.remove('btn-loading');
      delete buttonElement.dataset.originalContent;
    }
  }

  /**
   * Affiche/met à jour une progress bar
   * @param {string} selector - Sélecteur du conteneur
   * @param {number} percent - Pourcentage (0-100)
   * @param {string} label - Label optionnel
   */
  static showProgress(selector, percent, label = null) {
    const container = document.querySelector(selector);
    if (!container) return;

    let progressBar = container.querySelector('.progress-bar-custom');

    if (!progressBar) {
      container.innerHTML = `
        <div class="progress-bar-custom">
          <div class="progress-bar-fill"></div>
          <span class="progress-bar-text">0%</span>
        </div>
      `;
      progressBar = container.querySelector('.progress-bar-custom');
    }

    const fill = progressBar.querySelector('.progress-bar-fill');
    const text = progressBar.querySelector('.progress-bar-text');

    fill.style.width = `${percent}%`;
    text.textContent = label || `${Math.round(percent)}%`;

    if (percent >= 100) {
      setTimeout(() => {
        if (progressBar && progressBar.parentNode) {
          progressBar.remove();
        }
      }, 500);
    }
  }

  /**
   * Cache la progress bar
   * @param {string} selector - Sélecteur du conteneur
   */
  static hideProgress(selector) {
    const container = document.querySelector(selector);
    if (!container) return;

    const progressBar = container.querySelector('.progress-bar-custom');
    if (progressBar) {
      progressBar.remove();
    }
  }

  /**
   * Affiche un état d'erreur avec bouton retry
   * @param {string} selector - Sélecteur du conteneur
   * @param {Error|string} error - Erreur
   * @param {Function} retryCallback - Fonction de retry
   */
  static showErrorState(selector, error, retryCallback = null) {
    const container = document.querySelector(selector);
    if (!container) return;

    const errorMessage = error instanceof Error ? error.message : error;

    container.innerHTML = `
      <div class="error-state">
        <div class="error-state-icon">
          <i class="fas fa-exclamation-triangle fa-3x text-danger"></i>
        </div>
        <div class="error-state-content">
          <h5>Une erreur est survenue</h5>
          <p class="text-muted">${errorMessage}</p>
          ${retryCallback ? `
            <button class="btn btn-primary" id="retryButton">
              <i class="fas fa-redo"></i> Réessayer
            </button>
          ` : ''}
        </div>
      </div>
    `;

    container.classList.add('error-state-container');

    if (retryCallback) {
      const retryButton = container.querySelector('#retryButton');
      if (retryButton) {
        retryButton.addEventListener('click', retryCallback);
      }
    }
  }

  /**
   * Affiche un état vide (no data)
   * @param {string} selector - Sélecteur du conteneur
   * @param {string} message - Message
   * @param {string} icon - Classe d'icône FontAwesome
   * @param {Function} actionCallback - Action optionnelle
   * @param {string} actionLabel - Label du bouton d'action
   */
  static showEmptyState(selector, message = 'Aucune donnée disponible', icon = 'fa-inbox', actionCallback = null, actionLabel = 'Ajouter') {
    const container = document.querySelector(selector);
    if (!container) return;

    container.innerHTML = `
      <div class="empty-state">
        <div class="empty-state-icon">
          <i class="fas ${icon} fa-3x text-muted"></i>
        </div>
        <div class="empty-state-content">
          <p class="text-muted">${message}</p>
          ${actionCallback ? `
            <button class="btn btn-primary" id="emptyStateAction">
              <i class="fas fa-plus"></i> ${actionLabel}
            </button>
          ` : ''}
        </div>
      </div>
    `;

    container.classList.add('empty-state-container');

    if (actionCallback) {
      const actionButton = container.querySelector('#emptyStateAction');
      if (actionButton) {
        actionButton.addEventListener('click', actionCallback);
      }
    }
  }

  /**
   * Affiche un spinner fullscreen (overlay)
   * @param {string} message - Message optionnel
   */
  static showFullscreenLoader(message = 'Chargement...') {
    // Supprime loader existant
    this.hideFullscreenLoader();

    const loader = document.createElement('div');
    loader.id = 'fullscreenLoader';
    loader.className = 'fullscreen-loader';
    loader.innerHTML = `
      <div class="fullscreen-loader-content">
        <div class="spinner-border text-primary" role="status" style="width: 3rem; height: 3rem;">
          <span class="visually-hidden">Chargement...</span>
        </div>
        <p class="mt-3 mb-0">${message}</p>
      </div>
    `;

    document.body.appendChild(loader);

    // Empêche le scroll
    document.body.style.overflow = 'hidden';
  }

  /**
   * Cache le spinner fullscreen
   */
  static hideFullscreenLoader() {
    const loader = document.getElementById('fullscreenLoader');
    if (loader) {
      loader.remove();
    }

    // Restaure le scroll
    document.body.style.overflow = '';
  }

  /**
   * Wrapper pour exécuter une fonction async avec loading state
   * @param {string} selector - Sélecteur du conteneur
   * @param {Function} asyncFunction - Fonction async à exécuter
   * @param {Object} options - Options (type, rows, errorMessage)
   * @returns {Promise} Résultat de la fonction
   */
  static async withLoading(selector, asyncFunction, options = {}) {
    const {
      type = 'table',
      rows = 5,
      errorMessage = 'Erreur lors du chargement',
      retryCallback = null
    } = options;

    try {
      this.showSkeleton(selector, type, rows);
      const result = await asyncFunction();
      this.hideSkeleton(selector);
      return result;
    } catch (error) {
      console.error('[LoadingStateManager] Erreur:', error);
      this.showErrorState(selector, errorMessage, retryCallback);
      throw error;
    }
  }

  /**
   * Wrapper pour exécuter une fonction async avec bouton loading
   * @param {HTMLElement} buttonElement - Bouton
   * @param {Function} asyncFunction - Fonction async
   * @param {string} loadingText - Texte pendant chargement
   * @returns {Promise} Résultat de la fonction
   */
  static async withButtonLoading(buttonElement, asyncFunction, loadingText = 'Chargement...') {
    try {
      this.buttonLoading(buttonElement, true, loadingText);
      const result = await asyncFunction();
      this.buttonLoading(buttonElement, false);
      return result;
    } catch (error) {
      this.buttonLoading(buttonElement, false);
      throw error;
    }
  }
}

// Export pour utilisation dans les modules
export default LoadingStateManager;
