// ==========================================
// CRYPTOHELPER - CHIFFREMENT DES DONNÉES SENSIBLES
// AutoCalc OptiDevis v2.6.5
// ==========================================
/**
 * CONTEXTE DE SÉCURITÉ : ENCRYPTION AT REST
 *
 * Les mots de passe SMTP ne doivent JAMAIS être stockés en clair dans config.json.
 * Ce module utilise AES-256-CBC pour chiffrer/déchiffrer les mots de passe.
 *
 * PROTECTIONS IMPLEMENTÉES :
 * 1. Chiffrement AES-256-CBC (algorithme standard NSA)
 * 2. Clé dérivée avec scrypt (KDF sécurisé)
 * 3. IV aléatoire pour chaque chiffrement (empêche analyse)
 * 4. Gestion sécurisée de la clé de chiffrement
 *
 * IMPORTANT :
 * - La clé de chiffrement doit être stockée dans une variable d'environnement
 * - Ne JAMAIS commiter la clé dans Git
 * - Changer la clé si elle est compromise
 */

const crypto = require('crypto');

class CryptoHelper {
  /**
   * Obtient la clé de chiffrement depuis l'environnement ou génère une clé par défaut
   * IMPORTANT : En production, définir ENCRYPTION_KEY dans les variables d'environnement
   * @returns {Buffer} Clé de 32 bytes pour AES-256
   */
  static getEncryptionKey() {
    // Tenter de récupérer la clé depuis l'environnement
    const envKey = process.env.ENCRYPTION_KEY;

    if (envKey) {
      // Dériver une clé de 32 bytes depuis la clé d'environnement
      return crypto.scryptSync(envKey, 'optidevis-salt-v1', 32);
    }

    // FALLBACK : Clé par défaut (à changer en production)
    // Cette clé par défaut offre une sécurité basique si ENCRYPTION_KEY n'est pas définie
    const defaultKey = 'optidevis-default-encryption-key-change-in-production-2024';
    return crypto.scryptSync(defaultKey, 'optidevis-salt-v1', 32);
  }

  /**
   * Chiffre une chaîne de caractères
   * @param {string} text - Texte à chiffrer
   * @returns {string} - Texte chiffré au format "iv:encrypted"
   */
  static encrypt(text) {
    if (!text || typeof text !== 'string') {
      throw new Error('Le texte à chiffrer doit être une chaîne non vide');
    }

    try {
      const algorithm = 'aes-256-cbc';
      const key = this.getEncryptionKey();

      // Générer un IV aléatoire (16 bytes pour AES)
      const iv = crypto.randomBytes(16);

      // Créer le cipher
      const cipher = crypto.createCipheriv(algorithm, key, iv);

      // Chiffrer le texte
      let encrypted = cipher.update(text, 'utf8', 'hex');
      encrypted += cipher.final('hex');

      // Retourner IV + texte chiffré (séparés par :)
      return iv.toString('hex') + ':' + encrypted;
    } catch (error) {
      throw new Error(`Erreur de chiffrement: ${error.message}`);
    }
  }

  /**
   * Déchiffre une chaîne de caractères
   * @param {string} encryptedText - Texte chiffré au format "iv:encrypted"
   * @returns {string} - Texte déchiffré
   */
  static decrypt(encryptedText) {
    if (!encryptedText || typeof encryptedText !== 'string') {
      throw new Error('Le texte chiffré doit être une chaîne non vide');
    }

    try {
      const algorithm = 'aes-256-cbc';
      const key = this.getEncryptionKey();

      // Séparer IV et texte chiffré
      const parts = encryptedText.split(':');
      if (parts.length !== 2) {
        throw new Error('Format de texte chiffré invalide');
      }

      const iv = Buffer.from(parts[0], 'hex');
      const encrypted = parts[1];

      // Créer le decipher
      const decipher = crypto.createDecipheriv(algorithm, key, iv);

      // Déchiffrer le texte
      let decrypted = decipher.update(encrypted, 'hex', 'utf8');
      decrypted += decipher.final('utf8');

      return decrypted;
    } catch (error) {
      throw new Error(`Erreur de déchiffrement: ${error.message}`);
    }
  }

  /**
   * Vérifie si une chaîne est chiffrée (format iv:encrypted)
   * @param {string} text - Texte à vérifier
   * @returns {boolean} - true si le texte semble chiffré
   */
  static isEncrypted(text) {
    if (!text || typeof text !== 'string') {
      return false;
    }

    // Format attendu: 32 caractères hex + ':' + données chiffrées en hex
    const parts = text.split(':');
    if (parts.length !== 2) {
      return false;
    }

    // Vérifier que l'IV est bien 32 caractères hex (16 bytes)
    if (parts[0].length !== 32 || !/^[0-9a-f]+$/i.test(parts[0])) {
      return false;
    }

    // Vérifier que le texte chiffré est en hex
    if (!/^[0-9a-f]+$/i.test(parts[1])) {
      return false;
    }

    return true;
  }

  /**
   * Chiffre un objet de configuration SMTP
   * @param {Object} smtpConfig - Configuration SMTP avec mot de passe en clair
   * @returns {Object} - Configuration SMTP avec mot de passe chiffré
   */
  static encryptSMTPConfig(smtpConfig) {
    if (!smtpConfig || typeof smtpConfig !== 'object') {
      throw new Error('Configuration SMTP invalide');
    }

    // Copier la config pour ne pas modifier l'original
    const encrypted = { ...smtpConfig };

    // Chiffrer le mot de passe s'il existe et n'est pas déjà chiffré
    if (encrypted.pass && !this.isEncrypted(encrypted.pass)) {
      encrypted.pass = this.encrypt(encrypted.pass);
    }

    // Chiffrer également auth.pass si présent (structure nodemailer)
    if (encrypted.auth && encrypted.auth.pass && !this.isEncrypted(encrypted.auth.pass)) {
      encrypted.auth.pass = this.encrypt(encrypted.auth.pass);
    }

    return encrypted;
  }

  /**
   * Déchiffre un objet de configuration SMTP
   * @param {Object} smtpConfig - Configuration SMTP avec mot de passe chiffré
   * @returns {Object} - Configuration SMTP avec mot de passe en clair
   */
  static decryptSMTPConfig(smtpConfig) {
    if (!smtpConfig || typeof smtpConfig !== 'object') {
      throw new Error('Configuration SMTP invalide');
    }

    // Copier la config pour ne pas modifier l'original
    const decrypted = { ...smtpConfig };

    // Déchiffrer le mot de passe s'il est chiffré
    if (decrypted.pass && this.isEncrypted(decrypted.pass)) {
      try {
        decrypted.pass = this.decrypt(decrypted.pass);
      } catch (error) {
        console.error('Erreur de déchiffrement du mot de passe SMTP:', error.message);
        // Laisser le mot de passe chiffré si le déchiffrement échoue
      }
    }

    // Déchiffrer également auth.pass si présent
    if (decrypted.auth && decrypted.auth.pass && this.isEncrypted(decrypted.auth.pass)) {
      try {
        decrypted.auth.pass = this.decrypt(decrypted.auth.pass);
      } catch (error) {
        console.error('Erreur de déchiffrement du mot de passe SMTP (auth):', error.message);
      }
    }

    return decrypted;
  }

  /**
   * Génère une clé de chiffrement aléatoire sécurisée
   * Utile pour initialiser ENCRYPTION_KEY lors de la première installation
   * @returns {string} - Clé de chiffrement en hex (64 caractères)
   */
  static generateEncryptionKey() {
    return crypto.randomBytes(32).toString('hex');
  }
}

module.exports = CryptoHelper;
