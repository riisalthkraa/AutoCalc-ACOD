// AutoCalc OptiDevis v2.6.5

/**
 * CONTEXTE DE SÉCURITÉ : SERVER-SIDE VALIDATION
 *
 * La validation côté client (ValidationSystem.js) peut être contournée.
 * Ce module assure une validation STRICTE côté serveur avant toute
 * insertion/modification en base de données.
 *
 * PROTECTIONS IMPLEMENTÉES :
 * 1. Validation SIRET avec algorithme Luhn
 * 2. Validation email RFC 5322 compliant
 * 3. Validation téléphone français stricte
 * 4. Sanitization des chaînes pour prévenir XSS
 * 5. Validation des types et plages de valeurs
 */

class ServerValidator {
  /**
   * Valide les données d'un client
   * @param {Object} clientData - Données du client à valider
   * @returns {Object} { valid: boolean, errors: string[], data: Object }
   */
  static validateClient(clientData) {
    const errors = [];
    const sanitized = {};

    // Nom (obligatoire)
    if (!clientData.name || typeof clientData.name !== 'string' || clientData.name.trim().length < 2) {
      errors.push('Nom client requis (minimum 2 caractères)');
    } else if (clientData.name.length > 200) {
      errors.push('Nom client trop long (maximum 200 caractères)');
    } else {
      sanitized.name = this.sanitizeString(clientData.name);
    }

    // Contact (optionnel)
    if (clientData.contact) {
      if (typeof clientData.contact !== 'string' || clientData.contact.length > 200) {
        errors.push('Contact invalide (maximum 200 caractères)');
      } else {
        sanitized.contact = this.sanitizeString(clientData.contact);
      }
    }

    // Email (optionnel mais doit être valide si renseigné)
    if (clientData.email) {
      const emailResult = this.validateEmail(clientData.email);
      if (!emailResult.valid) {
        errors.push(emailResult.message);
      } else {
        sanitized.email = emailResult.value;
      }
    }

    // Téléphone (optionnel mais doit être valide si renseigné)
    if (clientData.phone) {
      const phoneResult = this.validatePhone(clientData.phone);
      if (!phoneResult.valid) {
        errors.push(phoneResult.message);
      } else {
        sanitized.phone = phoneResult.value;
      }
    }

    // Adresse (optionnel)
    if (clientData.address) {
      if (typeof clientData.address !== 'string' || clientData.address.length > 500) {
        errors.push('Adresse invalide (maximum 500 caractères)');
      } else {
        sanitized.address = this.sanitizeString(clientData.address);
      }
    }

    // Type (optionnel)
    if (clientData.type) {
      const validTypes = ['Particulier', 'Professionnel', 'Entreprise', 'Administration'];
      if (!validTypes.includes(clientData.type)) {
        errors.push(`Type client invalide (valeurs autorisées: ${validTypes.join(', ')})`);
      } else {
        sanitized.type = clientData.type;
      }
    }

    // SIRET (optionnel mais doit être valide si renseigné)
    if (clientData.siret) {
      const siretResult = this.validateSIRET(clientData.siret);
      if (!siretResult.valid) {
        errors.push(siretResult.message);
      } else {
        sanitized.siret = siretResult.value;
      }
    }

    // Notes (optionnel)
    if (clientData.notes) {
      if (typeof clientData.notes !== 'string' || clientData.notes.length > 1000) {
        errors.push('Notes trop longues (maximum 1000 caractères)');
      } else {
        sanitized.notes = this.sanitizeString(clientData.notes);
      }
    }

    // Copier l'ID si présent (pour les updates)
    if (clientData.id) {
      sanitized.id = String(clientData.id);
    }

    return {
      valid: errors.length === 0,
      errors,
      data: errors.length === 0 ? sanitized : null
    };
  }

  /**
   * Valide les données d'un fournisseur
   * @param {Object} supplierData - Données du fournisseur à valider
   * @returns {Object} { valid: boolean, errors: string[], data: Object }
   */
  static validateSupplier(supplierData) {
    const errors = [];
    const sanitized = {};

    // Nom (obligatoire)
    if (!supplierData.name || typeof supplierData.name !== 'string' || supplierData.name.trim().length < 2) {
      errors.push('Nom fournisseur requis (minimum 2 caractères)');
    } else if (supplierData.name.length > 200) {
      errors.push('Nom fournisseur trop long (maximum 200 caractères)');
    } else {
      sanitized.name = this.sanitizeString(supplierData.name);
    }

    // Contact (optionnel)
    if (supplierData.contact) {
      if (typeof supplierData.contact !== 'string' || supplierData.contact.length > 200) {
        errors.push('Contact invalide (maximum 200 caractères)');
      } else {
        sanitized.contact = this.sanitizeString(supplierData.contact);
      }
    }

    // Email (optionnel mais doit être valide si renseigné)
    if (supplierData.email) {
      const emailResult = this.validateEmail(supplierData.email);
      if (!emailResult.valid) {
        errors.push(emailResult.message);
      } else {
        sanitized.email = emailResult.value;
      }
    }

    // Téléphone (optionnel mais doit être valide si renseigné)
    if (supplierData.phone) {
      const phoneResult = this.validatePhone(supplierData.phone);
      if (!phoneResult.valid) {
        errors.push(phoneResult.message);
      } else {
        sanitized.phone = phoneResult.value;
      }
    }

    // Adresse (optionnel)
    if (supplierData.address) {
      if (typeof supplierData.address !== 'string' || supplierData.address.length > 500) {
        errors.push('Adresse invalide (maximum 500 caractères)');
      } else {
        sanitized.address = this.sanitizeString(supplierData.address);
      }
    }

    // Type (optionnel)
    if (supplierData.type) {
      const validTypes = ['Matériaux', 'Outillage', 'Services', 'Autre'];
      if (!validTypes.includes(supplierData.type)) {
        errors.push(`Type fournisseur invalide (valeurs autorisées: ${validTypes.join(', ')})`);
      } else {
        sanitized.type = supplierData.type;
      }
    }

    // SIRET (optionnel mais doit être valide si renseigné)
    if (supplierData.siret) {
      const siretResult = this.validateSIRET(supplierData.siret);
      if (!siretResult.valid) {
        errors.push(siretResult.message);
      } else {
        sanitized.siret = siretResult.value;
      }
    }

    // Délai livraison (optionnel)
    if (supplierData.deliveryDelay !== undefined && supplierData.deliveryDelay !== null && supplierData.deliveryDelay !== '') {
      const delay = parseInt(supplierData.deliveryDelay);
      if (isNaN(delay) || delay < 0 || delay > 365) {
        errors.push('Délai de livraison invalide (0-365 jours)');
      } else {
        sanitized.deliveryDelay = delay;
      }
    }

    // Site web (optionnel)
    if (supplierData.website) {
      if (typeof supplierData.website !== 'string' || supplierData.website.length > 200) {
        errors.push('Site web invalide (maximum 200 caractères)');
      } else if (!/^https?:\/\/.+/.test(supplierData.website)) {
        errors.push('Site web doit commencer par http:// ou https://');
      } else {
        sanitized.website = supplierData.website.trim();
      }
    }

    // Notes (optionnel)
    if (supplierData.notes) {
      if (typeof supplierData.notes !== 'string' || supplierData.notes.length > 1000) {
        errors.push('Notes trop longues (maximum 1000 caractères)');
      } else {
        sanitized.notes = this.sanitizeString(supplierData.notes);
      }
    }

    // Copier l'ID si présent (pour les updates)
    if (supplierData.id) {
      sanitized.id = String(supplierData.id);
    }

    return {
      valid: errors.length === 0,
      errors,
      data: errors.length === 0 ? sanitized : null
    };
  }

  /**
   * Valide un numéro SIRET avec l'algorithme Luhn
   * @param {string} siret - Numéro SIRET à valider
   * @returns {Object} { valid: boolean, message: string, value: string }
   */
  static validateSIRET(siret) {
    if (!siret || typeof siret !== 'string') {
      return { valid: false, message: 'SIRET invalide', value: null };
    }

    // Retirer les espaces
    const cleaned = siret.replace(/\s/g, '');

    // Vérifier 14 chiffres
    if (!/^\d{14}$/.test(cleaned)) {
      return { valid: false, message: 'SIRET doit contenir exactement 14 chiffres', value: null };
    }

    // Algorithme Luhn
    let sum = 0;
    for (let i = 0; i < 14; i++) {
      let digit = parseInt(cleaned[i]);
      if (i % 2 === 0) {
        digit *= 2;
        if (digit > 9) digit -= 9;
      }
      sum += digit;
    }

    if (sum % 10 !== 0) {
      return { valid: false, message: 'SIRET invalide (checksum Luhn incorrect)', value: null };
    }

    return { valid: true, message: '', value: cleaned };
  }

  /**
   * Valide une adresse email RFC 5322 compliant
   * @param {string} email - Email à valider
   * @returns {Object} { valid: boolean, message: string, value: string }
   */
  static validateEmail(email) {
    if (!email || typeof email !== 'string') {
      return { valid: false, message: 'Email invalide', value: null };
    }

    const trimmed = email.trim().toLowerCase();

    // Vérifier longueur max RFC 5321
    if (trimmed.length > 254) {
      return { valid: false, message: 'Email trop long (maximum 254 caractères)', value: null };
    }

    // RFC 5322 simplifié
    const emailRegex = /^[a-zA-Z0-9.!#$%&'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/;

    if (!emailRegex.test(trimmed)) {
      return { valid: false, message: 'Format email invalide', value: null };
    }

    return { valid: true, message: '', value: trimmed };
  }

  /**
   * Valide un numéro de téléphone français
   * @param {string} phone - Numéro de téléphone à valider
   * @returns {Object} { valid: boolean, message: string, value: string }
   */
  static validatePhone(phone) {
    if (!phone || typeof phone !== 'string') {
      return { valid: false, message: 'Téléphone invalide', value: null };
    }

    // Retirer tous les caractères non numériques sauf le +
    const cleaned = phone.replace(/[^\d+]/g, '');

    // Formats acceptés: +33XXXXXXXXX, 0033XXXXXXXXX, 0XXXXXXXXX
    let normalized;

    if (cleaned.startsWith('+33')) {
      if (cleaned.length !== 12 || !/^\+33[1-9]\d{8}$/.test(cleaned)) {
        return { valid: false, message: 'Téléphone invalide (format +33: 12 chiffres)', value: null };
      }
      normalized = cleaned;
    } else if (cleaned.startsWith('0033')) {
      if (cleaned.length !== 13 || !/^0033[1-9]\d{8}$/.test(cleaned)) {
        return { valid: false, message: 'Téléphone invalide (format 0033: 13 chiffres)', value: null };
      }
      normalized = '+33' + cleaned.substring(4);
    } else if (cleaned.startsWith('0')) {
      if (cleaned.length !== 10 || !/^0[1-9]\d{8}$/.test(cleaned)) {
        return { valid: false, message: 'Téléphone invalide (format 0X XX XX XX XX)', value: null };
      }
      normalized = '+33' + cleaned.substring(1);
    } else {
      return { valid: false, message: 'Téléphone doit commencer par 0, +33 ou 0033', value: null };
    }

    return { valid: true, message: '', value: normalized };
  }

  /**
   * Sanitise une chaîne de caractères pour prévenir XSS
   * @param {string} str - Chaîne à sanitiser
   * @returns {string} - Chaîne sanitisée
   */
  static sanitizeString(str) {
    if (typeof str !== 'string') return '';

    // Retirer les caractères dangereux HTML
    return str
      .replace(/[<>]/g, '') // Retirer < et >
      .trim() // Retirer espaces début/fin
      .replace(/\s+/g, ' '); // Normaliser espaces multiples
  }

  /**
   * Valide une configuration SMTP
   * @param {Object} smtpConfig - Configuration SMTP à valider
   * @returns {Object} { valid: boolean, errors: string[] }
   */
  static validateSMTPConfig(smtpConfig) {
    const errors = [];

    // Service (obligatoire)
    if (!smtpConfig.service || typeof smtpConfig.service !== 'string') {
      errors.push('Service SMTP requis');
    }

    // User (obligatoire, doit être un email)
    if (!smtpConfig.user) {
      errors.push('Email SMTP requis');
    } else {
      const emailResult = this.validateEmail(smtpConfig.user);
      if (!emailResult.valid) {
        errors.push('Email SMTP invalide');
      }
    }

    // Password (obligatoire, minimum 6 caractères)
    if (!smtpConfig.pass || typeof smtpConfig.pass !== 'string') {
      errors.push('Mot de passe SMTP requis');
    } else if (smtpConfig.pass.length < 6) {
      errors.push('Mot de passe SMTP trop court (minimum 6 caractères)');
    }

    return { valid: errors.length === 0, errors };
  }

  /**
   * Valide une configuration d'entreprise
   * @param {Object} config - Configuration à valider
   * @returns {Object} { valid: boolean, errors: string[] }
   */
  static validateCompanyConfig(config) {
    const errors = [];

    // Nom entreprise (obligatoire)
    if (!config.companyName || typeof config.companyName !== 'string' || config.companyName.trim().length < 2) {
      errors.push('Nom entreprise requis (minimum 2 caractères)');
    }

    // Adresse (obligatoire)
    if (!config.companyAddress || typeof config.companyAddress !== 'string' || config.companyAddress.trim().length < 5) {
      errors.push('Adresse entreprise requise (minimum 5 caractères)');
    }

    // Téléphone (obligatoire)
    if (!config.companyPhone) {
      errors.push('Téléphone entreprise requis');
    } else {
      const phoneResult = this.validatePhone(config.companyPhone);
      if (!phoneResult.valid) {
        errors.push('Téléphone entreprise invalide');
      }
    }

    // Email (obligatoire)
    if (!config.companyEmail) {
      errors.push('Email entreprise requis');
    } else {
      const emailResult = this.validateEmail(config.companyEmail);
      if (!emailResult.valid) {
        errors.push('Email entreprise invalide');
      }
    }

    // SIRET (obligatoire)
    if (!config.companySiret) {
      errors.push('SIRET entreprise requis');
    } else {
      const siretResult = this.validateSIRET(config.companySiret);
      if (!siretResult.valid) {
        errors.push('SIRET entreprise invalide');
      }
    }

    return { valid: errors.length === 0, errors };
  }
}

module.exports = ServerValidator;
