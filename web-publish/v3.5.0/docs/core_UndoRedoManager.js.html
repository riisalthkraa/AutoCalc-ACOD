/**

 * Gestionnaire universel d'annulation/r√©tablissement des actions.

 * @example

 *   description: 'Suppression mat√©riau "Ciment"',
 *   undo: async () => { await restoreMaterial(material); },
 *   redo: async () => { await deleteMaterial(material.id); }

 *
 * @example

 * @example

class UndoRedoManager {
  constructor() {
    /**
     * Stack des actions effectu√©es (can undo)
     * @type {Action[]}
     * @private
     */
    this.undoStack = [];

    /**
     * Stack des actions annul√©es (can redo)
     * @type {Action[]}
     * @private
     */
    this.redoStack = [];

    /**
     * Limite de l'historique (d√©faut: 50 actions)
     * @type {number}
     * @private
     */
    this.maxHistorySize = 50;

    /**
     * Mode debug
     * @type {boolean}
     * @private
     */
    this.debug = false;

    /**
     * Actions en cours d'ex√©cution (pour √©viter les boucles infinies)
     * @type {boolean}
     * @private
     */
    this.isExecuting = false;

    /**
     * Persistence activ√©e
     * @type {boolean}
     * @private
     */
    this.persistenceEnabled = true;

    /**
     * Batch mode (grouper plusieurs actions)
     * @type {boolean}
     * @private
     */
    this.batchMode = false;

    /**
     * Actions en attente (batch)
     * @type {Action[]}
     * @private
     */
    this.batchActions = [];

    console.log('üîÑ UndoRedoManager initialized');
  }

  /**
   * Enregistre une nouvelle action dans l'historique
   * @param {Object} action - L'action √† enregistrer
   * @param {string} action.type - Type d'action (ex: 'MATERIAL_DELETE')
   * @param {string} action.description - Description lisible (ex: 'Suppression mat√©riau "Ciment"')
   * @param {Function} action.undo - Fonction async d'annulation
   * @param {Function} action.redo - Fonction async de r√©tablissement
   * @param {Object} [action.metadata] - M√©tadonn√©es optionnelles
   */
  record(action) {
    // Validation
    if (!action || typeof action !== 'object') {
      console.error('[UndoRedoManager] Invalid action:', action);
      return;
    }

    if (!action.type || !action.undo || !action.redo) {
      console.error('[UndoRedoManager] Action must have type, undo, and redo:', action);
      return;
    }

    // Si on est en train d'ex√©cuter un undo/redo, ne pas enregistrer
    if (this.isExecuting) {
      return;
    }

    // Enrichir l'action
    const enrichedAction = {
      ...action,
      timestamp: Date.now(),
      id: this.generateId(),
      description: action.description || action.type
    };

    // Mode batch: accumuler les actions
    if (this.batchMode) {
      this.batchActions.push(enrichedAction);
      this.log('Action added to batch:', enrichedAction.description);
      return;
    }

    // Ajouter √† l'historique
    this.undoStack.push(enrichedAction);

    // Vider la redoStack (on ne peut plus redo apr√®s une nouvelle action)
    this.redoStack = [];

    // Limiter la taille de l'historique
    if (this.undoStack.length > this.maxHistorySize) {
      this.undoStack.shift(); // Retirer la plus ancienne action
    }

    this.log('Action recorded:', enrichedAction.description, `(${this.undoStack.length} in history)`);

    // Sauvegarder dans localStorage
    if (this.persistenceEnabled) {
      this.persist();
    }

    // √âmettre √©v√©nement
    if (window.eventBus) {
      window.eventBus.emit('undoredo:action-recorded', {
        action: enrichedAction,
        canUndo: this.canUndo(),
        canRedo: this.canRedo()
      });
    }
  }

  /**
   * Annule la derni√®re action (Ctrl+Z)
   * @returns {Promise<boolean>} True si l'annulation a r√©ussi
   */
  async undo() {
    if (!this.canUndo()) {
      this.log('Cannot undo - history is empty');
      if (window.NotificationSystem) {
        window.NotificationSystem.info('Aucune action √† annuler');
      }
      return false;
    }

    const action = this.undoStack.pop();
    this.isExecuting = true;

    try {
      this.log('Undoing:', action.description);

      // Ex√©cuter la fonction d'annulation
      await action.undo();

      // D√©placer l'action vers la redoStack
      this.redoStack.push(action);

      this.log('Undo successful:', action.description);

      // Notification avec bouton "R√©tablir"
      if (window.NotificationSystem) {
        window.NotificationSystem.show(
          `Action annul√©e: ${action.description}`,
          'success',
          5000,
          'Annulation',
          [{
            label: 'R√©tablir',
            onClick: () => this.redo()
          }]
        );
      }

      // Sauvegarder
      if (this.persistenceEnabled) {
        this.persist();
      }

      // √âmettre √©v√©nement
      if (window.eventBus) {
        window.eventBus.emit('undoredo:undo', {
          action,
          canUndo: this.canUndo(),
          canRedo: this.canRedo()
        });
      }

      return true;

    } catch (error) {
      console.error('[UndoRedoManager] Undo failed:', error);

      // Remettre l'action dans l'undoStack
      this.undoStack.push(action);

      if (window.NotificationSystem) {
        window.NotificationSystem.error(`Erreur lors de l'annulation: ${error.message}`);
      }

      return false;

    } finally {
      this.isExecuting = false;
    }
  }

  /**
   * R√©tablit la derni√®re action annul√©e (Ctrl+Y)
   * @returns {Promise<boolean>} True si le r√©tablissement a r√©ussi
   */
  async redo() {
    if (!this.canRedo()) {
      this.log('Cannot redo - redo stack is empty');
      if (window.NotificationSystem) {
        window.NotificationSystem.info('Aucune action √† r√©tablir');
      }
      return false;
    }

    const action = this.redoStack.pop();
    this.isExecuting = true;

    try {
      this.log('Redoing:', action.description);

      // Ex√©cuter la fonction de r√©tablissement
      await action.redo();

      // Remettre l'action dans l'undoStack
      this.undoStack.push(action);

      this.log('Redo successful:', action.description);

      // Notification
      if (window.NotificationSystem) {
        window.NotificationSystem.show(
          `Action r√©tablie: ${action.description}`,
          'success',
          3000,
          'R√©tablissement'
        );
      }

      // Sauvegarder
      if (this.persistenceEnabled) {
        this.persist();
      }

      // √âmettre √©v√©nement
      if (window.eventBus) {
        window.eventBus.emit('undoredo:redo', {
          action,
          canUndo: this.canUndo(),
          canRedo: this.canRedo()
        });
      }

      return true;

    } catch (error) {
      console.error('[UndoRedoManager] Redo failed:', error);

      // Remettre l'action dans la redoStack
      this.redoStack.push(action);

      if (window.NotificationSystem) {
        window.NotificationSystem.error(`Erreur lors du r√©tablissement: ${error.message}`);
      }

      return false;

    } finally {
      this.isExecuting = false;
    }
  }

  /**
   * V√©rifie si on peut annuler
   * @returns {boolean}
   */
  canUndo() {
    return this.undoStack.length > 0;
  }

  /**
   * V√©rifie si on peut r√©tablir
   * @returns {boolean}
   */
  canRedo() {
    return this.redoStack.length > 0;
  }

  /**
   * D√©marre le mode batch (grouper plusieurs actions)
   */
  startBatch() {
    this.batchMode = true;
    this.batchActions = [];
    this.log('Batch mode started');
  }

  /**
   * Termine le mode batch et enregistre toutes les actions group√©es
   * @param {string} description - Description du groupe d'actions
   */
  endBatch(description = 'Actions group√©es') {
    this.batchMode = false;

    if (this.batchActions.length === 0) {
      this.log('Batch ended with no actions');
      return;
    }

    // Cr√©er une action composite
    const compositeAction = {
      type: 'BATCH',
      description,
      timestamp: Date.now(),
      id: this.generateId(),
      actions: [...this.batchActions],
      undo: async () => {
        // Annuler toutes les actions dans l'ordre inverse
        for (let i = this.batchActions.length - 1; i >= 0; i--) {
          await this.batchActions[i].undo();
        }
      },
      redo: async () => {
        // R√©tablir toutes les actions dans l'ordre normal
        for (const action of this.batchActions) {
          await action.redo();
        }
      }
    };

    // Ajouter √† l'historique
    this.undoStack.push(compositeAction);
    this.redoStack = [];

    this.log(`Batch ended: ${this.batchActions.length} actions grouped as "${description}"`);
    this.batchActions = [];

    // Sauvegarder
    if (this.persistenceEnabled) {
      this.persist();
    }
  }

  /**
   * Annule le mode batch sans enregistrer
   */
  cancelBatch() {
    this.batchMode = false;
    this.batchActions = [];
    this.log('Batch cancelled');
  }

  /**
   * R√©cup√®re l'historique des actions (pour affichage UI)
   * @returns {Object[]} Historique avec infos lisibles
   */
  getHistory() {
    return {
      undo: this.undoStack.map(action => ({
        id: action.id,
        type: action.type,
        description: action.description,
        timestamp: action.timestamp,
        timeAgo: this.getTimeAgo(action.timestamp)
      })),
      redo: this.redoStack.map(action => ({
        id: action.id,
        type: action.type,
        description: action.description,
        timestamp: action.timestamp,
        timeAgo: this.getTimeAgo(action.timestamp)
      }))
    };
  }

  /**
   * Vide tout l'historique
   */
  clearHistory() {
    this.undoStack = [];
    this.redoStack = [];
    this.log('History cleared');

    if (this.persistenceEnabled) {
      this.clearPersistence();
    }

    if (window.eventBus) {
      window.eventBus.emit('undoredo:history-cleared');
    }
  }

  /**
   * Sauvegarde l'historique dans localStorage
   * @private
   */
  persist() {
    try {
      // On ne peut pas sauvegarder les fonctions undo/redo
      // On sauvegarde seulement les m√©tadonn√©es pour affichage UI
      const data = {
        undoStack: this.undoStack.map(a => ({
          id: a.id,
          type: a.type,
          description: a.description,
          timestamp: a.timestamp
        })),
        redoStack: this.redoStack.map(a => ({
          id: a.id,
          type: a.type,
          description: a.description,
          timestamp: a.timestamp
        }))
      };

      localStorage.setItem('undoRedoHistory', JSON.stringify(data));
    } catch (e) {
      console.error('[UndoRedoManager] Persistence failed:', e);
    }
  }

  /**
   * Charge l'historique depuis localStorage
   * Note: Les fonctions undo/redo ne peuvent pas √™tre restaur√©es
   * @private
   */
  loadPersistence() {
    try {
      const saved = localStorage.getItem('undoRedoHistory');
      if (saved) {
        const data = JSON.parse(saved);
        // On ne peut restaurer que les m√©tadonn√©es, pas les fonctions
        this.log('Loaded history metadata from localStorage');
      }
    } catch (e) {
      console.error('[UndoRedoManager] Load persistence failed:', e);
    }
  }

  /**
   * Supprime l'historique sauvegard√©
   * @private
   */
  clearPersistence() {
    try {
      localStorage.removeItem('undoRedoHistory');
    } catch (e) {
      console.error('[UndoRedoManager] Clear persistence failed:', e);
    }
  }

  /**
   * G√©n√®re un ID unique pour une action
   * @private
   * @returns {string}
   */
  generateId() {
    return `action_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }

  /**
   * Calcule le temps √©coul√© depuis une action
   * @private
   * @param {number} timestamp
   * @returns {string}
   */
  getTimeAgo(timestamp) {
    const seconds = Math.floor((Date.now() - timestamp) / 1000);

    if (seconds < 60) return `il y a ${seconds}s`;
    if (seconds < 3600) return `il y a ${Math.floor(seconds / 60)}min`;
    if (seconds < 86400) return `il y a ${Math.floor(seconds / 3600)}h`;
    return `il y a ${Math.floor(seconds / 86400)}j`;
  }

  /**
   * Active/d√©sactive le mode debug
   * @param {boolean} enabled
   */
  setDebug(enabled) {
    this.debug = enabled;
    console.log(`[UndoRedoManager] Debug mode: ${enabled ? 'ON' : 'OFF'}`);
  }

  /**
   * Active/d√©sactive la persistence
   * @param {boolean} enabled
   */
  setPersistence(enabled) {
    this.persistenceEnabled = enabled;
    console.log(`[UndoRedoManager] Persistence: ${enabled ? 'ON' : 'OFF'}`);
  }

  /**
   * D√©finit la limite de l'historique
   * @param {number} size
   */
  setMaxHistorySize(size) {
    this.maxHistorySize = size;
    console.log(`[UndoRedoManager] Max history size: ${size}`);

    // Ajuster l'historique si n√©cessaire
    while (this.undoStack.length > size) {
      this.undoStack.shift();
    }
  }

  /**
   * Log conditionnel (si debug activ√©)
   * @private
   */
  log(...args) {
    if (this.debug) {
      console.log('[UndoRedoManager]', ...args);
    }
  }
}
