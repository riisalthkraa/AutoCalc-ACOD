/**
 * FORMATTERS.JS - Utilitaires de formatage
 *
 * Fournit des fonctions de formatage pour :
 * - Dates (format français)
 * - Montants (format EUR)
 * - Texte (échappement HTML)
 *
 * @module formatters
 * @version 1.0.0
 * @author Claude AI - AutoCalc OptiDevis Refactoring
 */

/**
 * DateFormatter - Formatage des dates
 */
export const DateFormatter = {
  /**
   * Convertit une date en format français (JJ/MM/AAAA)
   * @param {string|Date} dateString - Date à formater
   * @returns {string} Date formatée en français
   */
  toFrench: (dateString) => {
    if (!dateString) return '';
    try {
      const date = new Date(dateString);
      if (isNaN(date.getTime())) return dateString;

      const day = String(date.getDate()).padStart(2, '0');
      const month = String(date.getMonth() + 1).padStart(2, '0');
      const year = date.getFullYear();
      return `${day}/${month}/${year}`;
    } catch (e) {
      return dateString;
    }
  }
};

/**
 * TextFormatter - Formatage du texte
 */
export const TextFormatter = {
  /**
   * Échappe les caractères HTML pour éviter les injections XSS
   * @param {string} text - Texte à échapper
   * @returns {string} Texte échappé
   */
  escapeHtml: (text) => {
    if (!text) return '';
    return String(text)
      .replace(/&/g, '&amp;')
      .replace(/</g, '&lt;')
      .replace(/>/g, '&gt;')
      .replace(/"/g, '&quot;')
      .replace(/'/g, '&#039;')
      .replace(/\//g, '&#x2F;');
  }
};

/**
 * CurrencyFormatter - Formatage des montants
 */
export const CurrencyFormatter = {
  /**
   * Formate un montant en euros (format français)
   * @param {number} amount - Montant à formater
   * @returns {string} Montant formaté (ex: "1 234,56 €")
   */
  format: (amount) => {
    if (amount === null || amount === undefined || isNaN(amount)) return '0,00 €';
    return new Intl.NumberFormat('fr-FR', {
      style: 'currency',
      currency: 'EUR'
    }).format(amount);
  }
};

// Exposer au scope global pour compatibilité avec l'ancien code
if (typeof window !== 'undefined') {
  window.DateFormatter = DateFormatter;
  window.TextFormatter = TextFormatter;
  window.CurrencyFormatter = CurrencyFormatter;
}
