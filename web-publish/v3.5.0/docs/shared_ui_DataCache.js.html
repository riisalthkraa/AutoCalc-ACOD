/**
 * DataCache - Système de cache avec TTL
 *
 * @module DataCache
 * @version 1.0.0
 */

import { logger } from '../utils/helpers.js';

class DataCache {
  constructor() {
    this.cache = new Map();
    this.stats = { hits: 0, misses: 0, sets: 0 };
  }

  /**
   * Ajoute une entrée au cache
   * @param {string} key - Clé
   * @param {*} value - Valeur
   * @param {number} ttl - Time to live en ms (défaut: 5 min)
   */
  set(key, value, ttl = 5 * 60 * 1000) {
    this.cache.set(key, {
      value,
      expires: Date.now() + ttl,
      createdAt: Date.now()
    });
    this.stats.sets++;
    logger.debug(`[Cache] SET: ${key} (TTL: ${ttl}ms)`);
  }

  /**
   * Récupère une entrée du cache
   * @param {string} key - Clé
   * @returns {*} Valeur ou null
   */
  get(key) {
    const entry = this.cache.get(key);

    if (!entry) {
      this.stats.misses++;
      logger.debug(`[Cache] MISS: ${key}`);
      return null;
    }

    if (Date.now() > entry.expires) {
      this.cache.delete(key);
      this.stats.misses++;
      logger.debug(`[Cache] EXPIRED: ${key}`);
      return null;
    }

    this.stats.hits++;
    logger.debug(`[Cache] HIT: ${key}`);
    return entry.value;
  }

  /**
   * Invalide une entrée du cache
   * @param {string} key - Clé
   */
  invalidate(key) {
    const deleted = this.cache.delete(key);
    if (deleted) {
      logger.info(`[Cache] INVALIDATE: ${key}`);
    }
  }

  /**
   * Invalide toutes les entrées correspondant à un pattern
   * @param {string} pattern - Pattern à rechercher
   */
  invalidatePattern(pattern) {
    let count = 0;
    for (const key of this.cache.keys()) {
      if (key.includes(pattern)) {
        this.cache.delete(key);
        count++;
      }
    }
    if (count > 0) {
      logger.info(`[Cache] INVALIDATE PATTERN "${pattern}": ${count} entries removed`);
    }
  }

  /**
   * Vide tout le cache
   */
  clear() {
    const size = this.cache.size;
    this.cache.clear();
    logger.info(`[Cache] CLEAR: ${size} entries removed`);
  }

  /**
   * Récupère les statistiques du cache
   * @returns {Object} Statistiques
   */
  getStats() {
    const hitRate = this.stats.hits + this.stats.misses > 0
      ? ((this.stats.hits / (this.stats.hits + this.stats.misses)) * 100).toFixed(1)
      : 0;

    return {
      ...this.stats,
      size: this.cache.size,
      hitRate: `${hitRate}%`
    };
  }
}

// Instance singleton
const dataCache = new DataCache();

export default dataCache;
