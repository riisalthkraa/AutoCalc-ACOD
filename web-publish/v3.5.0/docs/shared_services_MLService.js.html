/**
 * MLService - Machine Learning API Integration
 *
 * Service pour int√©grer l'API ML Python (optionnel en DEV)
 *
 * IMPORTANT:
 * - En DEV: Utilise l'API Python si disponible (localhost:5000)
 * - En PROD: Fallback vers algorithmes JavaScript simples
 * - Python NON requis pour la distribution finale
 *
 * @module MLService
 */

const ML_API_URL = 'http://localhost:5000';
const ML_API_TIMEOUT = 3000; // 3 secondes timeout

class MLService {
    constructor() {
        this.mlAvailable = false;
        this.cache = new Map(); // Cache pour optimiser performances
        this.lastHealthCheck = null;
        this.healthCheckInterval = 60000; // 1 minute
    }

    /**
     * V√©rifie si le serveur ML est disponible
     * @returns {Promise<boolean>}
     */
    async checkMLServerAvailable() {
        try {
            const controller = new AbortController();
            const timeoutId = setTimeout(() => controller.abort(), 1000); // 1s timeout

            const response = await fetch(`${ML_API_URL}/api/health`, {
                method: 'GET',
                signal: controller.signal
            });

            clearTimeout(timeoutId);

            if (response.ok) {
                this.mlAvailable = true;
                this.lastHealthCheck = Date.now();
                console.log('‚úÖ Serveur ML disponible - Features ML activ√©es');
                return true;
            }
        } catch (error) {
            // Serveur ML non disponible (normal en PROD)
            this.mlAvailable = false;
        }

        console.log('‚ö†Ô∏è  Serveur ML non disponible - Mode fallback activ√©');
        return false;
    }

    /**
     * V√©rifie p√©riodiquement la disponibilit√© du serveur ML
     * @returns {Promise<boolean>}
     */
    async ensureMLAvailable() {
        // Re-v√©rifier uniquement si dernier check > 1 minute
        if (!this.lastHealthCheck || Date.now() - this.lastHealthCheck > this.healthCheckInterval) {
            await this.checkMLServerAvailable();
        }
        return this.mlAvailable;
    }

    /**
     * Sugg√®re des mat√©riaux similaires
     * @param {string} materialName - Nom du mat√©riau recherch√©
     * @param {number} limit - Nombre de suggestions (d√©faut: 5)
     * @returns {Promise<Array>}
     */
    async suggestMaterials(materialName, limit = 5) {
        const cacheKey = `suggest_${materialName}_${limit}`;

        // V√©rifier cache
        if (this.cache.has(cacheKey)) {
            return this.cache.get(cacheKey);
        }

        // Essayer API ML
        if (await this.ensureMLAvailable()) {
            try {
                const response = await fetch(`${ML_API_URL}/api/suggest-materials`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ material_name: materialName, limit }),
                    signal: AbortSignal.timeout(ML_API_TIMEOUT)
                });

                if (response.ok) {
                    const data = await response.json();
                    this.cache.set(cacheKey, data.suggestions);
                    return data.suggestions;
                }
            } catch (error) {
                console.warn('‚ö†Ô∏è  API ML erreur, fallback vers algorithme simple:', error.message);
            }
        }

        // Fallback: Algorithme JavaScript simple
        return await this._suggestMaterialsFallback(materialName, limit);
    }

    /**
     * Fallback: Suggestions basiques en JavaScript
     * @private
     */
    async _suggestMaterialsFallback(materialName, limit) {
        try {
            // R√©cup√©rer tous les mat√©riaux depuis window.materials (d√©j√† charg√©s)
            const materials = window.materials || [];
            const searchLower = materialName.toLowerCase();

            // Recherche textuelle simple
            const matches = materials
                .map(material => {
                    const name = (material.name || '').toLowerCase();
                    const desc = (material.description || '').toLowerCase();
                    const category = (material.category || '').toLowerCase();
                    const searchText = `${name} ${desc} ${category}`;

                    // Calcul de similarit√© simple
                    let similarity = 0;
                    if (name.includes(searchLower)) similarity += 0.8;
                    else if (searchText.includes(searchLower)) similarity += 0.4;

                    // Mots communs
                    const searchWords = searchLower.split(' ');
                    searchWords.forEach(word => {
                        if (word.length > 2 && searchText.includes(word)) {
                            similarity += 0.1;
                        }
                    });

                    return {
                        ...material,
                        similarity: Math.min(similarity, 1.0)
                    };
                })
                .filter(m => m.similarity > 0.1)
                .sort((a, b) => b.similarity - a.similarity)
                .slice(0, limit);

            return matches;
        } catch (error) {
            console.error('‚ùå Erreur suggestions fallback:', error);
            return [];
        }
    }

    /**
     * D√©tecte les anomalies de prix dans un devis
     * @param {number} price - Prix du devis
     * @param {number} itemsCount - Nombre d'items
     * @returns {Promise<Object>}
     */
    async detectPriceAnomalies(price, itemsCount) {
        // Essayer API ML
        if (await this.ensureMLAvailable()) {
            try {
                const response = await fetch(`${ML_API_URL}/api/detect-price-anomalies`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ price, items_count: itemsCount }),
                    signal: AbortSignal.timeout(ML_API_TIMEOUT)
                });

                if (response.ok) {
                    return await response.json();
                }
            } catch (error) {
                console.warn('‚ö†Ô∏è  API ML erreur, fallback:', error.message);
            }
        }

        // Fallback: D√©tection simple
        return await this._detectPriceAnomaliesFallback(price, itemsCount);
    }

    /**
     * Fallback: D√©tection anomalies basique
     * @private
     */
    async _detectPriceAnomaliesFallback(price, itemsCount) {
        try {
            // R√©cup√©rer historique depuis window.appState (d√©j√† charg√©)
            const quotes = window.appState?.get('quotes') || [];

            // Filtrer devis similaires
            const similar = quotes.filter(q =>
                Math.abs((q.items || []).length - itemsCount) <= 2
            );

            if (similar.length < 5) {
                return { is_anomaly: false, warning: 'Pas assez de donn√©es' };
            }

            // Calcul statistiques simples
            const prices = similar.map(q => q.total || 0);
            const mean = prices.reduce((a, b) => a + b, 0) / prices.length;
            const variance = prices.reduce((a, b) => a + Math.pow(b - mean, 2), 0) / prices.length;
            const std = Math.sqrt(variance);

            // Z-score
            const zScore = Math.abs((price - mean) / std);
            const isAnomaly = zScore > 2;

            let warning = null;
            if (isAnomaly) {
                const percentDiff = Math.round((price - mean) / mean * 100);
                if (price > mean) {
                    warning = `Prix anormalement √©lev√© (+${percentDiff}%)`;
                } else {
                    warning = `Prix anormalement bas (${percentDiff}%)`;
                }
            }

            return {
                is_anomaly: isAnomaly,
                z_score: Math.round(zScore * 100) / 100,
                expected_range: [
                    Math.round(mean - 2 * std),
                    Math.round(mean + 2 * std)
                ],
                mean_price: Math.round(mean),
                warning: warning,
                samples_count: similar.length
            };

        } catch (error) {
            console.error('‚ùå Erreur d√©tection anomalies fallback:', error);
            return { is_anomaly: false, warning: null };
        }
    }

    /**
     * Pr√©dit le prix optimal pour un devis
     * @param {number} itemsCount - Nombre d'items
     * @param {string} clientType - Type de client (standard, premium, new)
     * @returns {Promise<Object>}
     */
    async predictQuotePrice(itemsCount, clientType = 'standard') {
        // Essayer API ML
        if (await this.ensureMLAvailable()) {
            try {
                const response = await fetch(`${ML_API_URL}/api/predict-quote-price`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ items_count: itemsCount, client_type: clientType }),
                    signal: AbortSignal.timeout(ML_API_TIMEOUT)
                });

                if (response.ok) {
                    return await response.json();
                }
            } catch (error) {
                console.warn('‚ö†Ô∏è  API ML erreur, fallback:', error.message);
            }
        }

        // Fallback: Pr√©diction simple
        return await this._predictQuotePriceFallback(itemsCount, clientType);
    }

    /**
     * Fallback: Pr√©diction basique
     * @private
     */
    async _predictQuotePriceFallback(itemsCount, clientType) {
        try {
            const quotes = window.appState?.get('quotes') || [];
            const similar = quotes.filter(q =>
                Math.abs((q.items || []).length - itemsCount) <= 3
            );

            if (similar.length < 3) {
                return { error: 'Pas assez de donn√©es historiques' };
            }

            const prices = similar.map(q => q.total || 0);
            const avgPrice = prices.reduce((a, b) => a + b, 0) / prices.length;
            const std = Math.sqrt(
                prices.reduce((a, b) => a + Math.pow(b - avgPrice, 2), 0) / prices.length
            );

            // Ajuster selon type client
            const multipliers = { standard: 1.0, premium: 1.15, new: 0.95 };
            const multiplier = multipliers[clientType] || 1.0;
            const predicted = avgPrice * multiplier;

            return {
                predicted_price: Math.round(predicted),
                confidence_interval: [
                    Math.round(predicted - std),
                    Math.round(predicted + std)
                ],
                samples_count: similar.length,
                client_type: clientType
            };

        } catch (error) {
            console.error('‚ùå Erreur pr√©diction fallback:', error);
            return { error: 'Erreur calcul' };
        }
    }

    /**
     * Recommande les clients √† relancer
     * @returns {Promise<Array>}
     */
    async recommendFollowup() {
        // Essayer API ML
        if (await this.ensureMLAvailable()) {
            try {
                const response = await fetch(`${ML_API_URL}/api/recommend-followup`, {
                    method: 'GET',
                    signal: AbortSignal.timeout(ML_API_TIMEOUT)
                });

                if (response.ok) {
                    const data = await response.json();
                    return data.recommendations || [];
                }
            } catch (error) {
                console.warn('‚ö†Ô∏è  API ML erreur, fallback:', error.message);
            }
        }

        // Fallback: Recommandations simples
        return await this._recommendFollowupFallback();
    }

    /**
     * Fallback: Recommandations basiques
     * @private
     */
    async _recommendFollowupFallback() {
        try {
            // R√©cup√©rer donn√©es depuis window.appState (d√©j√† charg√©es)
            const clients = window.appState?.get('clients') || [];
            const quotes = window.appState?.get('quotes') || [];
            const now = Date.now();

            const recommendations = [];

            for (const client of clients) {
                const clientQuotes = quotes.filter(q => q.client_id === client.id);

                if (clientQuotes.length > 0) {
                    // Derni√®re activit√©
                    const lastDate = Math.max(...clientQuotes.map(q => new Date(q.created_at).getTime()));
                    const daysSince = Math.floor((now - lastDate) / (1000 * 60 * 60 * 24));

                    // Valeur totale
                    const totalValue = clientQuotes.reduce((sum, q) => sum + (q.total || 0), 0);

                    // Score priorit√©
                    const priorityScore = (daysSince / 30) * (totalValue / 1000);

                    if (daysSince > 60 && totalValue > 500) {
                        recommendations.push({
                            client_id: client.id,
                            client_name: client.name,
                            days_since_last: daysSince,
                            total_value: Math.round(totalValue),
                            priority_score: Math.round(priorityScore * 100) / 100,
                            quotes_count: clientQuotes.length,
                            reason: `Inactif depuis ${daysSince} jours, ${Math.round(totalValue)}‚Ç¨ de revenus`
                        });
                    }
                }
            }

            // Trier par priorit√©
            recommendations.sort((a, b) => b.priority_score - a.priority_score);

            return recommendations.slice(0, 10);

        } catch (error) {
            console.error('‚ùå Erreur recommandations fallback:', error);
            return [];
        }
    }

    /**
     * Efface le cache
     */
    clearCache() {
        this.cache.clear();
        console.log('üóëÔ∏è  Cache ML effac√©');
    }
}
